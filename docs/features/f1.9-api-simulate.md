# F1.9: POST /api/simulate — Weapon × Target → JSON
**Цель (SMART):** Реализовать POST /api/simulate с Pydantic-моделями, поиском юнитов/оружия, rate limiter (10/час Free, 100/час Premium). Критерий: curl запрос возвращает JSON с mean_damage, pmf_by_damage, kill_probability; невалидный weapon_id → 400; превышение лимита → 429. 2 часа.

**Файлы:**
- `web/routes/api.py` (расширить)
- `backend/api/simulate.py` (создать — логика эндпоинта)

**Зависимости:** F1.6 (Combat), F1.4 (Wiki Loader), F1.3 (Modifiers)
**Часы:** 2h

## API Contract

### Request

```http
POST /api/simulate
Content-Type: application/json
Authorization: Bearer <jwt>

{
```

### Errors

| HTTP | Body |
|------|------|
| 400 | `{"error": "weapon_not_found", "detail": "Shoota not in registry"}` |
| 400 | `{"error": "unit_not_found", "detail": "Tactical Marine not in registry"}` |
| 401 | `{"error": "unauthorized"}` |
| 429 | `{"error": "rate_limit_exceeded", "detail": "Max 10 simulations/hour on Free tier"}` |
| 500 | `{"error": "simulation_failed", "detail": "..."}` |

## Pydantic Models

```python
from pydantic import BaseModel, Field
from typing import Literal, Optional

class SimulateRequest(BaseModel):
    weapon_id: str = Field(..., min_length=1)
    attacker_id: str = Field(..., min_length=1)
    defender_id: str = Field(..., min_length=1)
    n_simulations: int = Field(default=10000, ge=100, le=100000)
    modifiers: list[str] = Field(default_factory=list)
    range_bracket: Literal["long", "half"] = "long"

class StepBreakdown(BaseModel):
    attacks: float
    hits: float
    wounds: float
    unsaved_wounds: float
    damage_dealt: float

class PMFEntry(BaseModel):
    damage: int
    probability: float

class SimulateResult(BaseModel):
    mean_damage: float
    median_damage: float
    std_damage: float
    min_damage: int
    max_damage: int
    kill_probability: float
    wound_distribution: list[float]
    pmf_by_damage: list[PMFEntry]
    steps: StepBreakdown
    simulation_time_ms: float

class SimulateResponse(BaseModel):
    weapon: str
    attacker: str
    defender: str
    n_simulations: int
    seed: int
    results: SimulateResult
    meta: dict
```

## Handler

```python
from fastapi import APIRouter, Depends, HTTPException
from backend.auth import get_current_user
from backend.loader.registry import registry
from backend.engine.combat import simulate_weapon_attack
from backend.engine.dice import DicePool
from backend.engine.modifiers import Modifier

router = APIRouter(prefix="/api", tags=["simulation"])

MODIFIER_MAP = {
    "heavy": Modifier("hit_roll", "add", 1, source="ability",
                       condition={"stationary": True}),
    "lethal_hits": Modifier("hit_roll", "lethal_hits", source="ability"),
    "sustained_hits_1": Modifier("hit_roll", "sustained_hits", 1, source="ability"),
    # ...
}

@router.post("/simulate", response_model=SimulateResponse)
async def simulate(
    req: SimulateRequest,
    user = Depends(get_current_user),
):
    # Проверка rate limit
    if not user.tier == "premium":
        sim_count = get_recent_sim_count(user.id, window_minutes=60)
        if sim_count >= 10:
            raise HTTPException(429, detail="Max 10 simulations/hour on Free tier")

    # Загрузка registry
    units = registry.load()
    if not units:
        raise HTTPException(503, detail="Wiki not loaded")

    # Поиск юнитов
    attacker_name = _find_unit_name(req.attacker_id, units)
    defender_name = _find_unit_name(req.defender_id, units)

    if not attacker_name:
        raise HTTPException(400, detail=f"Unit '{req.attacker_id}' not found")
    if not defender_name:
        raise HTTPException(400, detail=f"Unit '{req.defender_id}' not found")

    attacker = units[attacker_name]
    defender = units[defender_name]

    # Поиск оружия
    weapon = _find_weapon(req.weapon_id, attacker)
    if not weapon:
        raise HTTPException(400, detail=f"Weapon '{req.weapon_id}' not found on {attacker.name}")

    # Build modifiers
    modifiers = []
    for mod_key in req.modifiers:
        if mod_key in MODIFIER_MAP:
            modifiers.append(MODIFIER_MAP[mod_key])

    # Симуляция
    pool = DicePool(seed=42)
    result = simulate_weapon_attack(
        weapon=weapon,
        attacker=attacker,
        defender=defender,
        modifiers=modifiers,
        pool=pool,
        n_iterations=req.n_simulations,
    )

    # Логирование для rate limit
    log_simulation(user.id)

    return _to_response(result, weapon, attacker, defender, req.n_simulations)


def _find_unit_name(query: str, units: dict[str, Unit]) -> Optional[str]:
    """Поиск юнита: точное совпадение → case-insensitive → contains."""
    query_lower = query.lower()
    for name in units:
        if name.lower() == query_lower:
            return name
    for name in units:
        if query_lower in name.lower():
            return name
    return None


def _find_weapon(query: str, unit: Unit) -> Optional[Weapon]:
    """Поиск оружия на юните."""
    query_lower = query.lower()
    for w in unit.ranged_weapons:
        if w.name.lower() == query_lower:
            return w
    for w in unit.melee_weapons:
        if w.name.lower() == query_lower:
            return w
    return None


def _to_response(result, weapon, attacker, defender, n):
    """CombatResult → SimulateResponse."""
    return SimulateResponse(
        weapon=weapon.name,
        attacker=attacker.name,
        defender=defender.name,
        n_simulations=n,
        seed=42,
        results=SimulateResult(
            mean_damage=result.stats.mean,
            median_damage=result.stats.median,
            std_damage=result.stats.std,
            min_damage=result.stats.min_val,
            max_damage=result.stats.max_val,
            kill_probability=result.stats.kill_probability,
            wound_distribution=result.stats.wound_distribution,
            pmf_by_damage=[
                PMFEntry(damage=d, probability=p)
                for d, p in result.stats.pmf
            ],
            steps=StepBreakdown(
                attacks=result.avg_attacks,
                hits=result.avg_hits,
                wounds=result.avg_wounds,
                unsaved_wounds=result.avg_unsaved_wounds,
                damage_dealt=result.stats.mean,
            ),
            simulation_time_ms=result.simulation_time_ms,
        ),
        meta={"engine_version": "1.0", "edition": "10e"},
    )
```

## Rate Limiting

```python
from collections import defaultdict
from datetime import datetime, timedelta

# In-memory rate limiter (P1 — потом перенести в БД)
_sim_log: dict[int, list[datetime]] = defaultdict(list)

def get_recent_sim_count(user_id: int, window_minutes: int = 60) -> int:
    cutoff = datetime.utcnow() - timedelta(minutes=window_minutes)
    return sum(1 for t in _sim_log[user_id] if t > cutoff)

def log_simulation(user_id: int) -> None:
    _sim_log[user_id].append(datetime.utcnow())
    # Очистка старых записей
    cutoff = datetime.utcnow() - timedelta(hours=2)
    _sim_log[user_id] = [t for t in _sim_log[user_id] if t > cutoff]
```

```python
def test_simulate_endpoint(client):
    resp = client.post("/api/simulate", json={
        "weapon_id": "shoota",
        "attacker_id": "boyz",
        "defender_id": "tactical_marine",
        "n_simulations": 1000,

    assert resp.status_code == 200
    data = resp.json()
    assert "results" in data
    assert data["results"]["mean_damage"] > 0
    assert "pmf_by_damage" in data["results"]

def test_simulate_unknown_weapon(client):

        "defender_id": "tactical_marine",
    })
    assert resp.status_code == 400
```
