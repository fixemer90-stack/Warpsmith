# F3.1: Greedy Decision Engine — Target/Action Evaluation

**Файл:** `backend/engine/ai/decision.py` (создать)
**Зависимости:** F2.1 (Game State), F2.3 (LoS), F1.6 (Combat), F2.5 (Game Loop)
**Часы:** 6h

**Цель (SMART):** Реализовать жадный decision engine, который для заданного UnitState вычисляет наилучшее действие (цель, оружие, режим стрельбы/движения) на основе текущего GameState. Критерий: engine возвращает Action с максимальным score для любого допустимого состояния (минимум 6 тестов, покрывающих приоритизацию целей, выбор оружия, оценку движения и edge cases). 6 часов.

class ActionType(Enum):

@dataclass
class Action:
    """Одно действие, оценённое decision engine."""
    type: ActionType
    target_id: Optional[str] = None          # ID цели (юнит или позиция)
    weapon_index: Optional[int] = None       # какое оружие использовать
    mode: Optional[str] = None               # "normal", "overcharge", "grenade"

    state: GameState                          # полное состояние игры
    opponent_units: list[UnitState]           # список вражеских юнитов (UnitState)
    phase: GamePhase                          # фаза хода (enum, не str)
    turn: int
    opponent_units_map: dict[str, Unit] = field(default_factory=dict)  # unit_id → Unit model
    rng: np.random.Generator = field(default_factory=lambda: np.random.default_rng())
    weights: dict[str, float] = field(default_factory=lambda: dict(DEFAULT_WEIGHTS))

## Decision Flow

```python
def choose_action(
    actor: UnitState,
    actor_unit: Unit,
    ctx: EvaluationContext,
) -> Action:
    """Главная точка входа: выбрать лучшее действие для юнита в фазе.

    Flow:
    1. Сгенерировать список кандидатов (все возможные действия)
    2. Оценить каждое через evaluate_action
    3. Выбрать с максимальным score
    4. Вернуть Action
    """
    candidates = _generate_candidates(actor, actor_unit, ctx)
    if not candidates:
        return Action(type=ActionType.HOLD, score=0.0,
                       rationale="no actions generated")

    results = [evaluate_action(actor, actor_unit, a, ctx) for a in candidates]
    best = max(results, key=lambda a: a.score)
    return best


def _generate_candidates(
    actor: UnitState,
    actor_unit: Unit,
    ctx: EvaluationContext,
) -> list[Action]:
    """Сгенерировать все возможные действия для юнита в текущей фазе."""
    candidates = []

    if ctx.phase == GamePhase.SHOOTING:
        if actor.is_engaged or actor.is_battle_shocked:
            return [Action(type=ActionType.HOLD, rationale="cannot shoot")]
        for wi, weapon in enumerate(actor_unit.ranged_weapons):
            for target in ctx.opponent_units:
                if not target.is_alive:
                    continue
                if not _in_weapon_range(actor, target, weapon):
                    continue
                if not _has_valid_los(actor, target, ctx.state):
                    continue
                candidates.append(Action(
                    type=ActionType.SHOOT,
                    target_id=target.unit_id,
                    weapon_index=wi,
                    mode="normal",
                ))

    elif ctx.phase == GamePhase.CHARGE:
        if actor.is_battle_shocked:
            return [Action(type=ActionType.HOLD, rationale="battle-shocked")]
        for target in ctx.opponent_units:
            if not target.is_alive:
                continue
            dist = _distance(actor.position, target.position)
            if dist <= 12:  # max charge range
                candidates.append(Action(
                    type=ActionType.CHARGE,
                    target_id=target.unit_id,
                ))

    elif ctx.phase == GamePhase.MOVEMENT:
        # Двигаться к целям / от угрозы
        for target in ctx.opponent_units:
            if not target.is_alive:
                continue
            candidates.append(Action(
                type=ActionType.MOVE,
                target_position=target.position,
            ))
            candidates.append(Action(
                type=ActionType.ADVANCE,
                target_position=target.position,
            ))

    if not candidates:
        candidates.append(Action(type=ActionType.HOLD, rationale="no valid actions"))
    return candidates
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Все цели уничтожены | Вернуть HOLD action |
| Юнит engaged, фаза shooting | Не генерировать SHOOT кандидатов |
| Юнит advancing без Assault | Не генерировать SHOOT (кроме Assault weapons) |
| Нет целей в LoS | HOLD или MOVE для улучшения позиции |
| Distance до цели > max charge range | Не генерировать CHARGE |
| Цель с FNP/Invul | Учесть в expected_damage |
| Overkill (expected >> remaining) | Штраф, выбрать другое оружие/цель |
| Оружие с Blast | Бонус против больших отрядов (squad_size >= 6) |
| Оружие с Torrent | Авто-hit, не зависит от BS модификаторов |
| Обнуление весов (все 0) | Все действия имеют score 0, выбрать первое |

## Integration

```python
# Использование в Game Loop (f2.5) с реальным GameState API
from backend.state.game_state import GamePhase, GameState
from backend.model.unit import Unit
from backend.engine.ai.decision import (
    EvaluationContext, choose_action, ActionType
)


def shooting_phase(state: GameState, unit_models: dict[str, Unit]):
    """
    Провести shooting phase для active_player:
      - Определить оппонента (противоположный player_id)
      - Собрать opponent_units из state.players[opp_id].units
      - Собрать opponent_units_map из unit_models
      - Для каждого живого юнита active_player вызвать choose_action
    """
    # Определяем активного игрока и оппонента
    player_ids = list(state.players.keys())
    if len(player_ids) < 2:
        return
    active_id = player_ids[0] if state.players[player_ids[0]].command_priority else player_ids[1]
    opp_id = player_ids[1] if active_id == player_ids[0] else player_ids[0]

    active_player = state.players[active_id]
    opp_player = state.players[opp_id]

    # Собираем список opponent-юнитов и их моделей
    opponent_units = [u for u in opp_player.units.values() if u.is_alive]
    opponent_units_map = {
        uid: unit_models[uid]
        for uid in opp_player.units
        if uid in unit_models
    }

    for unit_state in active_player.units.values():
        if not unit_state.is_alive:
            continue
        if unit_state.has_shot:
            continue

        ctx = EvaluationContext(
            actor=unit_state,
            actor_unit=unit_models.get(unit_state.unit_id),
            state=state,
            opponent_units=opponent_units,
            phase=GamePhase.SHOOTING,
            turn=state.current_round,
            opponent_units_map=opponent_units_map,
        )
        action = choose_action(unit_state, ctx.actor_unit, ctx)
        if action.type == ActionType.SHOOT:
            damage = state.deal_damage(action.target_id, 1)  # упрощённо
            state.game_log.append(
                f"{unit_state.name} → {action.target_id} "
                f"(weapon={action.weapon_index}, score={action.score:.2f})"
            )
```
