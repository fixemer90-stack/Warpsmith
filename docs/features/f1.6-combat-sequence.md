```python
def simulate_weapon_attack(
    weapon: Weapon,
    attacker: Unit,
    defender: Unit,
    modifiers: Optional[list[Modifier]] = None,
    pool: Optional[DicePool] = None,
    n_iterations: int = 10000,
    distance: Optional[int] = None,
    is_stationary: bool = False,
    squad_size: int = 1,
) -> CombatResult:
    """
    Monte Carlo симуляция: оружие × цель.

    Возвращает CombatResult с массивом урона и статистиками.
    """
    if pool is None:
        pool = DicePool(seed=42)

    all_mods = list(modifiers or []) + build_weapon_modifiers(weapon)

    t0 = time.monotonic()

    damage = pool.simulate(
        lambda rng: _single_attack_sequence(
            rng, weapon, attacker, defender, all_mods,
            distance, is_stationary, squad_size,
        ),
        n=n_iterations,
    )

    t1 = time.monotonic()

    # Для шагов — усреднённый проход (не Monte Carlo)
    avg_hits, avg_wounds, avg_unsaved = _expected_steps(
        weapon, attacker, defender, all_mods, distance, is_stationary, squad_size
    )
    attacks = _resolve_attacks(weapon, squad_size)

    stats = compute_stats(damage, target_wounds=defender.wounds)

    return CombatResult(
        damage_per_iter=damage,
        stats=stats,
        avg_attacks=attacks,
        avg_hits=avg_hits,
        avg_wounds=avg_wounds,
        avg_unsaved_wounds=avg_unsaved,
        avg_damage_per_success=stats.mean / max(avg_unsaved, 0.001),
        n_iterations=n_iterations,
        weapon_name=weapon.name,
        target_name=defender.name,
        simulation_time_ms=(t1 - t0) * 1000,
    )


def _single_attack_sequence(
    rng: np.random.Generator,
    weapon: Weapon,
    attacker: Unit,
    defender: Unit,
    modifiers: list[Modifier],
    distance: Optional[int],
    is_stationary: bool,
    squad_size: int,
) -> int:
    """Один прогон: все атаки оружия → общий урон."""

    ctx = ModifierContext(
        attacker=attacker,
        defender=defender,
        weapon=weapon,
        distance=distance,
        is_stationary=is_stationary,
        squad_size=squad_size,
    )

    # 1. Количество атак
    num_attacks = _resolve_attacks(weapon, squad_size, modifiers, ctx, rng)

    total_damage = 0
    for _ in range(num_attacks):
        # 2. Hit Roll
        hit_target = weapon.skill
        hit_mod = apply_modifiers("hit_roll", hit_target, modifiers, ctx, rng)
        hit_roll = _roll_with_modifiers(rng, hit_mod, modifiers, "hit_roll", ctx)

        if not hit_roll.success:
            continue

        # Sustained Hits: доп. атаки на crit
        total_damage += _handle_sustained(weapon, rng, hit_mod, ctx, modifiers)

        # 3. Wound Roll
        wound_target = attacker.effective_toughness(defender.toughness)
        wound_mod = apply_modifiers("wound_roll", wound_target, modifiers, ctx, rng)
        wound_roll = _roll_with_modifiers(rng, wound_mod, modifiers, "wound_roll", ctx)

        if not wound_roll.success:
            continue

        # Lethal Hits bypass wound → auto wound
        if wound_mod.auto_success:
            pass  # wound already succeeded

        # 4. Save Roll
        save_target = defender.best_save(weapon.ap)
        if wound_mod.ignore_save:  # Devastating Wounds
            saves_failed = True
        else:
            save_mod = apply_modifiers("save_roll", save_target, modifiers, ctx, rng)
            save_roll = _roll_with_modifiers(rng, save_mod, modifiers, "save_roll", ctx)
            saves_failed = not save_roll.success

        if not saves_failed:
            continue

        # 5. Damage
        damage_dealt = resolve_dice(weapon.damage_dice, rng)

        # 6. FNP
        if defender.feel_no_pain:
            fnp_rolls = rng.integers(1, 7, size=damage_dealt)
            fnp_saves = int(np.sum(fnp_rolls >= defender.feel_no_pain))
            damage_dealt -= fnp_saves

        total_damage += max(0, damage_dealt)

    return total_damage


@dataclass
class RollResult:
    success: bool
    roll: int
    is_crit: bool = False


def _roll_with_modifiers(
    rng: np.random.Generator,
    mod_result: Any,  # ModifierResult
    modifiers: list[Modifier],
    step: str,
    ctx: ModifierContext,
) -> RollResult:
    """Бросок с учётом reroll и auto-success."""
    if mod_result.auto_success:
        return RollResult(success=True, roll=6, is_crit=True)

    roll = int(rng.integers(1, 7))

    # Reroll
    if should_reroll(roll, mod_result.target_value, modifiers, step):
        roll = int(rng.integers(1, 7))

    success = roll >= mod_result.target_value
    is_crit = roll == 6

    # Natural 1 всегда fail, natural 6 всегда success
    if roll == 1:
        success = False
    elif roll == 6:
        success = True

    return RollResult(success=success, roll=roll, is_crit=is_crit)
```

## Expected Values (deterministic)

```python
def _expected_steps(
    weapon, attacker, defender, modifiers,
    distance, is_stationary, squad_size,
) -> tuple[float, float, float]:
    """Ожидаемые средние значения для UI (без Monte Carlo)."""

    num_attacks = _resolve_attacks(weapon, squad_size)
    if num_attacks == 0:
        return (0.0, 0.0, 0.0)

    hit_prob = max(0, (7 - weapon.skill) / 6)
    wound_target = attacker.effective_toughness(defender.toughness)
    wound_prob = max(0, (7 - wound_target) / 6)
    save_target = defender.best_save(weapon.ap)
    save_prob = max(0, (7 - save_target) / 6)

    avg_hits = num_attacks * hit_prob
    avg_wounds = avg_hits * wound_prob
    avg_unsaved = avg_wounds * (1 - save_prob)

    return (avg_hits, avg_wounds, avg_unsaved)


def _resolve_attacks(
    weapon: Weapon,
    squad_size: int = 1,
    modifiers: list = None,
    ctx=None,
    rng=None,
) -> int:
    """Количество атак с учётом Blast."""
    if rng:
        count = resolve_dice(weapon.attacks_dice, rng)
    else:
        num, sides, bonus = weapon.attacks_dice
        count = bonus if num == 0 else round((num * (sides + 1) / 2) + bonus)

    # Blast
    if "blast" in weapon.tags and squad_size >= 5:
        if squad_size >= 10:
            count += 6  # or max attacks
        else:
            count += 1

    return max(0, count)
```

## Edge Cases

| Ситуация | Поведение |
|----------|-----------|
| Torrent weapon | auto-hit, пропустить hit roll |
| Lethal Hits + Devastating Wounds | hit=6 auto-wound, wound=6 ignore save |
| Damage > defender wounds | overflow не переносится |
| Natural 1 всегда fail | hit/wound/save |
| Natural 6 всегда success | hit/wound |
| Blast + Torrent | атаки считаются до auto-hit |
| FNP на 1 wound модели | модели с 1 раной: FNP не имеет смысла |
