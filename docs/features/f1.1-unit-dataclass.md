## Что добавить

```python
@dataclass
class Unit:
    # ... существующие поля ...
    invulnerable_save: Optional[int] = None       # 4 → 4++
    feel_no_pain: Optional[int] = None            # 5 → 5+++
    wargear_options: list[WargearSlot] = field(default_factory=list)
    transports: list[str] = field(default_factory=list)  # что можно перевозить

@dataclass
class WargearSlot:
    slot_name: str                                # "ranged_1", "melee_1"
    choices: list[str]                            # ["shoota", "big_shoota"]
    default_index: int = 0                        # индекс выбора по умолчанию
```

## Методы

```python
def effective_toughness(self, strength: int) -> int:
    """Вернуть target для wound roll по таблице S vs T."""
    # S >= 2*T → 2+; S > T → 3+; S == T → 4+; S < T → 5+; S <= T/2 → 6+

def best_save(self, ap: int) -> int:
    """Вернуть лучший сейв с учётом AP и invul."""
    modified_sv = self.save + ap  # AP отрицательный
    if self.invulnerable_save and self.invulnerable_save < modified_sv:
        return self.invulnerable_save
    return max(2, min(6, modified_sv))  # natural 1 always fails

def max_wounds_in_squad(self, squad_size: int) -> int:
    return self.wounds * squad_size
```

## Тесты

```python
def test_effective_toughness():
    marine = Unit(name="Tactical Marine", toughness=4, ...)
    assert marine.effective_toughness(8) == 2   # S >= 2*T
    assert marine.effective_toughness(5) == 3   # S > T
    assert marine.effective_toughness(4) == 4   # S == T
    assert marine.effective_toughness(3) == 5   # S < T
    assert marine.effective_toughness(2) == 6   # S <= T/2

def test_best_save_with_invul():
    terminator = Unit(name="Terminator", save=2, invulnerable_save=4, ...)
    assert terminator.best_save(-3) == 4  # invul лучше (2-(-3)=5 vs 4)
    assert terminator.best_save(-1) == 3  # save лучше (2-(-1)=3 vs 4)

def test_max_wounds():
    marine = Unit(name="Tactical Marine", wounds=2, ...)
    assert marine.max_wounds_in_squad(5) == 10
    assert marine.max_wounds_in_squad(10) == 20
```
