## Battle-shock

```python
import numpy as np

def battle_shock_test(unit: UnitState, rng: np.random.Generator) -> bool:
    """Battle-shock test: 2D6 >= LD.

    Returns True if unit becomes battle-shocked.
    Only units below half-strength test.
    """
    if unit.is_above_half_strength:
        return False

    roll = int(rng.integers(1, 7)) + int(rng.integers(1, 7))

    # Natural 1 on either die = auto-fail
    # Natural 6 on both = auto-pass
    if roll <= 2:  # snake eyes
        unit.is_battle_shocked = True
        return True
    if roll >= 12:  # boxcars
        unit.is_battle_shocked = False
        return False

    # Get LD from unit data
    unit_data = registry.get_unit(unit.unit_name)
    ld = unit_data.leadership if unit_data else 7

    if roll < ld:
        unit.is_battle_shocked = True
        return True
    else:
        unit.is_battle_shocked = False
        return False
```

## CP Generation

```python
def apply_cp_generation(state: GameState) -> GameState:
    """Command Phase: каждому игроку +1 CP (или +2 с Warlord)."""
    for player_id, roster in [(1, state.roster_a), (2, state.roster_b)]:
        cp_gain = 1
        if roster.warlord_id:
            cp_gain = 2
        roster.cp += cp_gain

        # Leviathan cap: max 10 CP per player
        roster.cp = min(roster.cp, 10)

    return state
```

## Stratagem Framework

```python
from dataclasses import dataclass, field
from typing import Callable, Optional

@dataclass
class Stratagem:
    name: str
    cp_cost: int
    phase: str          # "any", "command", "movement", "shooting", "charge", "fight"
    description: str
    effect_fn: Callable  # (state, target, kwargs) -> state
    condition_fn: Optional[Callable] = None  # (state, target) -> bool

class StratagemRegistry:
    """Реестр всех стратагем."""
    def __init__(self):
        self._stratagems: dict[str, Stratagem] = {}

    def register(self, stratagem: Stratagem) -> None:
        self._stratagems[stratagem.name] = stratagem

    def get(self, name: str) -> Optional[Stratagem]:
        return self._stratagems.get(name)

    def available(self, state: GameState, phase: str) -> list[Stratagem]:
        """Список стратагем, доступных в данной фазе."""
        return [s for s in self._stratagems.values()
                if s.phase in ("any", phase)
                and (not s.condition_fn or s.condition_fn(state, None))]

    def execute(self, state: GameState, name: str,
                target=None, **kwargs) -> GameState:
        """Применить стратагему."""
        s = self.get(name)
        if s is None:
            raise ValueError(f"Stratagem '{name}' not found")
        roster = state.active_roster()
        if roster.cp < s.cp_cost:
            raise ValueError(f"Not enough CP: need {s.cp_cost}, have {roster.cp}")
        roster.cp -= s.cp_cost
        return s.effect_fn(state, target, **kwargs)

# Global registry
stratagem_registry = StratagemRegistry()
```

def _init_core_stratagems():

    ))

    s.register(Stratagem(

        phase="fight",
        description="Fight first in Fight phase after opponent activation.",
        effect_fn=lambda state, target, **kw: _counter_offensive(state, target),

    ))

def test_battle_shock_below_half():
    unit = UnitState(squad_size=5, wounds_remaining=4,
                     is_above_half_strength=False, ...)
    shocked = battle_shock_test(unit, np.random.default_rng(42))
    assert isinstance(shocked, bool)
