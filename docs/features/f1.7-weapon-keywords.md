# F1.7: Weapon Keywords — Sustained, Lethal, Devastating, Ignores Cover
**Цель (SMART):** Реализовать TAG_TO_MODIFIERS для Sustained Hits (1/2/3), Lethal Hits, Devastating Wounds, Ignores Cover, Torrent, Twin-linked, Blast, Anti-keyword. Критерий: crit на hit=6 даёт extra_attacks, crit на wound=6 игнорирует save, Torrent пропускает hit roll, Blast даёт +1 атаку при ≥5 моделях. 3 часа.

def build_weapon_modifiers(weapon: Weapon) -> list[Modifier]:

    mods: list[Modifier] = []

        mods.extend(TAG_TO_MODIFIERS.get(tag, []))

## Critical Hit handling

```python
def handle_critical_hit(
    roll_result: RollResult,
    step: str,
    modifiers: list[Modifier],
    ctx: ModifierContext,
) -> CriticalEffect:
    """Обработать natural 6 (или anti-keyword crit) на hit/wound шаге."""
    if step == "hit_roll" and roll_result.is_crit:
        has_lethal = any(
            m.operation == "lethal_hits" for m in modifiers if m.target == "hit_roll"
        )
        sustained_count = 0
        for m in modifiers:
            if m.target == "hit_roll" and m.operation == "sustained_hits":
                sustained_count = m.value

        return CriticalEffect(
            auto_wound=has_lethal,
            extra_attacks=sustained_count,
        )

    if step == "wound_roll" and roll_result.is_crit:
        has_devastating = any(
            m.operation == "devastating_wounds" for m in modifiers
            if m.target == "wound_roll"
        )
        return CriticalEffect(
            ignore_save=has_devastating,
        )

    return CriticalEffect()
```

## Anti-Keyword (X+ vs Keyword)

```python
@dataclass
class AntiKeyword:
    """Anti-Infantry 4+: threshold=4, target_keyword="Infantry""""
    threshold: int      # wound on X+
    target_keyword: str  # Infantry, Vehicle, Monster, etc.

def parse_anti_tag(tag: str) -> Optional[AntiKeyword]:
    """Парсинг 'anti_infantry_4' → AntiKeyword(4, 'Infantry')."""
    m = re.match(r'anti_(\w+)_(\d+)', tag)
    if m:
        kw = m.group(1).title()  # infantry → Infantry
        threshold = int(m.group(2))
        return AntiKeyword(threshold, kw)
    return None

def resolve_anti_wound(
    roll: int,
    anti: AntiKeyword,
    defender: Unit,
) -> bool:
    """
    Anti-keyword делает wound roll на X+ против цели с keyword.

    Если roll >= threshold и у цели есть keyword → critical wound.
    Critical wound не обязательно означает success — это доп. эффект.
    """
    if anti.target_keyword in defender.keywords:
        if roll >= anti.threshold:
            return True  # critical wound
    return False
```

## Integration in Combat Sequence

```python
# Вставка в _single_attack_sequence (combat.py):

# --- Hit step ---
hit_roll = _roll_with_modifiers(...)

if not hit_roll.success:
    continue

# Critical effects on hit
hit_crit = handle_critical_hit(hit_roll, "hit_roll", modifiers, ctx)
if hit_crit.auto_wound:
    wound_success = True  # Lethal Hits bypass wound
if hit_crit.extra_attacks:
    for _ in range(hit_crit.extra_attacks):
        # Extra attacks auto-hit, proceed to wound
        ...

# --- Wound step ---
if not wound_success:
    wound_roll = _roll_with_modifiers(...)
    wound_crit = handle_critical_hit(wound_roll, "wound_roll", modifiers, ctx)

    if wound_crit.ignore_save:
        ignore_save = True  # Devastating Wounds bypass save

# --- Save step ---
if ignore_save:
    saves_failed = True  # skip save roll
```
