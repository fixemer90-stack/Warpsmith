## Constants

```python
import numpy as np
from dataclasses import dataclass, field
from typing import Optional

# Terrain types
TERRAIN_OPEN = 0
TERRAIN_LIGHT_COVER = 1       # Obscuring, benefit of cover
TERRAIN_HEAVY_COVER = 2       # Dense, benefit of cover, -1 to hit
TERRAIN_IMPASSABLE = 3        # Wall, crater
TERRAIN_DEPLOY_ZONE_A = 10    # Player 1 deploy zone
TERRAIN_DEPLOY_ZONE_B = 11    # Player 2 deploy zone
TERRAIN_OBJECTIVE = 20        # Capture point

@dataclass
class TerrainFeature:
    name: str
    x: int
    y: int
    width: int
    height: int
    terrain_type: int
    is_blocking_los: bool = True

@dataclass
class GameMap:
    rows: int = 60
    cols: int = 48
    grid: np.ndarray = None
    los_cache: dict = field(default_factory=dict)
    features: list[TerrainFeature] = field(default_factory=list)

    def __post_init__(self):
        if self.grid is None:
            self.grid = np.zeros((self.rows, self.cols), dtype=np.int8)

    def get(self, x: int, y: int) -> int:
        if 0 <= x < self.cols and 0 <= y < self.rows:
            return int(self.grid[y, x])
        return TERRAIN_IMPASSABLE

    def is_passable(self, x: int, y: int) -> bool:
        return self.get(x, y) not in (TERRAIN_IMPASSABLE,)

    def has_cover(self, x: int, y: int) -> bool:
        t = self.get(x, y)
        return t in (TERRAIN_LIGHT_COVER, TERRAIN_HEAVY_COVER)

    def is_deploy_zone(self, x: int, y: int, player: int) -> bool:
        t = self.get(x, y)
        if player == 1:
            return t == TERRAIN_DEPLOY_ZONE_A
        return t == TERRAIN_DEPLOY_ZONE_B
```

## Map Builder

```python
def create_default_map() -> GameMap:
    """Стандартная карта 48x60 с симметричными deploy зонами."""
    m = GameMap(rows=60, cols=48)
    # Deploy zone A: rows 0-14
    m.grid[0:15, :] = TERRAIN_DEPLOY_ZONE_A
    # Deploy zone B: rows 45-59
    m.grid[45:60, :] = TERRAIN_DEPLOY_ZONE_B
    # Objectives: 3 точки
    m.grid[15, 8] = TERRAIN_OBJECTIVE
    m.grid[30, 24] = TERRAIN_OBJECTIVE
    m.grid[44, 40] = TERRAIN_OBJECTIVE
    # Terrain features
    m.features = [
        TerrainFeature("Ruin", x=12, y=20, width=4, height=4,
                       terrain_type=TERRAIN_LIGHT_COVER),
        TerrainFeature("Wall", x=30, y=15, width=2, height=10,
                       terrain_type=TERRAIN_IMPASSABLE),
        TerrainFeature("Forest", x=20, y=35, width=6, height=6,
                       terrain_type=TERRAIN_HEAVY_COVER),
    ]
    for f in m.features:
        _apply_feature(m, f)
    return m

def _apply_feature(m: GameMap, f: TerrainFeature) -> None:
    for dy in range(f.height):
        for dx in range(f.width):
            x, y = f.x + dx, f.y + dy
            if 0 <= x < m.cols and 0 <= y < m.rows:
                m.grid[y, x] = f.terrain_type
```

## JSON Config

```python
def map_from_config(config: dict) -> GameMap:
    """Создать карту из JSON-конфига."""
    m = GameMap(rows=config.get("rows", 60), cols=config.get("cols", 48))
    # Apply deploy zones
    for zone in config.get("deploy_zones", []):
        y_start = zone["y_start"]
        y_end = zone["y_end"]
        value = TERRAIN_DEPLOY_ZONE_A if zone["player"] == 1 else TERRAIN_DEPLOY_ZONE_B
        m.grid[y_start:y_end, :] = value
    # Apply features
    for feat_data in config.get("features", []):
        feat = TerrainFeature(**feat_data)
        m.features.append(feat)
        _apply_feature(m, feat)
    # Objectives
    for obj in config.get("objectives", []):
        m.grid[obj["y"], obj["x"]] = TERRAIN_OBJECTIVE
    return m
```

## Тесты

```python
def test_map_default_structure():
    m = create_default_map()
    assert m.grid.shape == (60, 48)
    assert m.get(24, 7) == TERRAIN_DEPLOY_ZONE_A
    assert m.get(24, 30) == TERRAIN_OPEN
    assert m.get(24, 50) == TERRAIN_DEPLOY_ZONE_B

def test_passable():
    m = create_default_map()
    assert m.is_passable(0, 0)
    assert not m.is_passable(30, 15)  # wall

def test_deploy_zone():
    m = create_default_map()
    assert m.is_deploy_zone(10, 5, 1)
    assert not m.is_deploy_zone(10, 5, 2)

def test_map_from_config():
    config = {
        "rows": 40, "cols": 30,
        "deploy_zones": [{"player": 1, "y_start": 0, "y_end": 10}],
        "features": [],
        "objectives": [{"x": 15, "y": 20}],
    }
    m = map_from_config(config)
    assert m.rows == 40
    assert m.cols == 30
    assert m.get(15, 20) == TERRAIN_OBJECTIVE
```
