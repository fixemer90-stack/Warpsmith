# F1.5: Dice Pool — NumPy D6 Monte Carlo
**Цель (SMART):** Создать DicePool на NumPy с методами d6(), d3(), successes(), simulate(). Критерий: 10 000 итераций < 50ms, seed=42 даёт идентичные результаты, simulate() возвращает np.ndarray, тесты покрывают range и reproducibility. 2 часа.

## Core API

```python
import numpy as np
from typing import Callable, Optional, Any

class DicePool:
    """Пул D6 для Monte Carlo симуляции."""

    def __init__(self, seed: Optional[int] = None):
        self.rng = np.random.default_rng(seed)

    def d6(self, count: int = 1) -> np.ndarray:
        """Бросок count D6. Возвращает массив int от 1 до 6."""
        return self.rng.integers(1, 7, size=count)

    def d3(self, count: int = 1) -> np.ndarray:
        """Бросок count D3. Результаты 1, 2, 3."""
        return self.rng.integers(1, 4, size=count)

    def successes(self, rolls: np.ndarray, target: int) -> int:
        """Сколько бросков >= target?"""
        return int(np.sum(rolls >= target))

    def roll_and_count(self, count: int, target: int) -> int:
        """Бросить count D6, вернуть сколько >= target."""
        return self.successes(self.d6(count), target)

    # --------------- Monte Carlo ---------------

    def simulate(
        self,
        func: Callable[[np.random.Generator], int],
        n: int = 10000,
    ) -> np.ndarray:
        """
        Запустить func N раз → массив результатов.

        func принимает rng и возвращает int (урон за одну итерацию).

        Использование:
            pool = DicePool(seed=42)
            results = pool.simulate(lambda rng: my_attack(rng, weapon, target))
        """
        return np.array([func(self.rng) for _ in range(n)])

    def simulate_batched(
        self,
        func: Callable[[np.random.Generator], int],
        n: int = 10000,
        batch_size: int = 1000,
    ) -> np.ndarray:
        """
        Версия с прогрессом для долгих симуляций.
        Можно добавить callback: func(rng, batch_i, n_batches).
        """
        results = []
        for i in range(0, n, batch_size):
            batch_n = min(batch_size, n - i)
            for _ in range(batch_n):
                results.append(func(self.rng))
        return np.array(results)
```

## Result Aggregation

```python
@dataclass
class SimulationStats:
    mean: float
    median: float
    std: float
    min_val: int
    max_val: int
    percentiles: dict[int, float]        # {25: 0.0, 50: 1.0, 75: 2.0, 95: 4.0}
    pmf: list[tuple[int, float]]         # [(0, 0.78), (1, 0.15), (2, 0.07)]
    kill_probability: float              # P(damage >= target_wounds)

def compute_stats(results: np.ndarray, target_wounds: int = 1) -> SimulationStats:
    """Вычислить статистики по массиву результатов Monte Carlo."""
    unique, counts = np.unique(results, return_counts=True)
    probs = counts / len(results)

    return SimulationStats(
        mean=float(np.mean(results)),
        median=float(np.median(results)),
        std=float(np.std(results)),
        min_val=int(np.min(results)),
        max_val=int(np.max(results)),
        percentiles={
            p: float(np.percentile(results, p))
            for p in [5, 25, 50, 75, 95]
        },
        pmf=[(int(d), float(p)) for d, p in zip(unique, probs)],
        kill_probability=float(np.mean(results >= target_wounds)),
    )
```

## Тесты

```python
def test_d6_range():
    pool = DicePool(seed=42)
    rolls = pool.d6(1000)
    assert np.all(rolls >= 1) and np.all(rolls <= 6)
    assert len(rolls) == 1000

def test_successes():
    pool = DicePool(seed=42)
    rolls = np.array([3, 4, 5, 6])
    assert pool.successes(rolls, 4) == 3  # 4, 5, 6

def test_seed_reproducibility():
    a = DicePool(seed=42).d6(10)
    b = DicePool(seed=42).d6(10)
    assert np.array_equal(a, b)

def test_simulate():
    pool = DicePool(seed=42)
    def always_one(_rng):
        return 1
    results = pool.simulate(always_one, n=100)
    assert np.all(results == 1)
    assert len(results) == 100

def test_compute_stats():
    results = np.array([0, 0, 0, 1, 1, 2])
    stats = compute_stats(results, target_wounds=2)
    assert abs(stats.mean - 0.67) < 0.01
    assert stats.pmf[0] == (0, 0.5)  # 3/6 = 0.5
    assert abs(stats.kill_probability - 1/6) < 0.01  # only 2 kills
```
