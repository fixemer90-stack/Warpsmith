## backup.py

```python
#!/usr/bin/env python3
"""
SQLite backup с ротацией: ежедневные (7), еженедельные (4).

Использует sqlite3 backup API (online backup без блокировки).
Сжимает gzip. Сохраняет в BACKUP_DIR.
"""

import os
import sys
import gzip
import shutil
import sqlite3
from pathlib import Path
from datetime import datetime, timedelta

BACKUP_DIR = Path(os.getenv("BACKUP_DIR", "/var/backups/warpsmith"))
DB_PATH = Path(os.getenv("DB_PATH", "simulator.db"))
RETENTION_DAILY = int(os.getenv("BACKUP_RETENTION_DAILY", "7"))
RETENTION_WEEKLY = int(os.getenv("BACKUP_RETENTION_WEEKLY", "4"))


def backup_database() -> Path:
    """Создать backup базы данных. Возвращает путь к файлу."""
    BACKUP_DIR.mkdir(parents=True, exist_ok=True)

    today = datetime.now()
    date_str = today.strftime("%Y-%m-%d")
    is_sunday = today.weekday() == 6  # Воскресенье — weekly backup

    # Временный файл для sqlite3 backup
    backup_path = BACKUP_DIR / f"simulator-{date_str}.sqlite"
    if is_sunday:
        backup_path = BACKUP_DIR / f"simulator-weekly-{date_str}.sqlite"

    # Online backup через sqlite3 backup API
    src_conn = sqlite3.connect(str(DB_PATH))
    dst_conn = sqlite3.connect(str(backup_path))
    with dst_conn:
        src_conn.backup(dst_conn)
    src_conn.close()
    dst_conn.close()

    # Сжать gzip
    gz_path = backup_path.with_suffix(".sqlite.gz")
    with open(backup_path, "rb") as f_in:
        with gzip.open(gz_path, "wb") as f_out:
            shutil.copyfileobj(f_in, f_out)

    # Удалить несжатый файл
    backup_path.unlink()

    return gz_path


def rotate_backups() -> None:
    """Удалить старые backup-файлы согласно retention."""
    now = datetime.now()

    # Ежедневные — удалить старше RETENTION_DAILY дней
    daily_files = sorted(BACKUP_DIR.glob("simulator-????-??-??.sqlite.gz"))
    for f in daily_files[:-RETENTION_DAILY] if len(daily_files) > RETENTION_DAILY else []:
        f.unlink()
        print(f"  Удалён старый daily backup: {f.name}")

    # Еженедельные — удалить старше RETENTION_WEEKLY недель
    weekly_files = sorted(BACKUP_DIR.glob("simulator-weekly-*.sqlite.gz"))
    for f in weekly_files[:-RETENTION_WEEKLY] if len(weekly_files) > RETENTION_WEEKLY else []:
        f.unlink()
        print(f"  Удалён старый weekly backup: {f.name}")


def main() -> None:
    print(f"Backup: {DB_PATH} → {BACKUP_DIR}")
    if not DB_PATH.exists():
        print(f"Ошибка: {DB_PATH} не найден", file=sys.stderr)
        sys.exit(1)

    gz_path = backup_database()
    print(f"  Создан: {gz_path.name} ({gz_path.stat().st_size / 1024:.1f} KB)")

    rotate_backups()
    print("  Ротация выполнена")
    print("✓ Backup завершён")


if __name__ == "__main__":
    main()
```

## restore.py

```python
#!/usr/bin/env python3
"""
Восстановление SQLite из backup.

Использование:
  python scripts/restore.py                    # последний backup
  python scripts/restore.py --date 2026-05-01  # конкретная дата
  python scripts/restore.py --file backup.sqlite.gz  # конкретный файл
"""

import os
import sys
import gzip
import shutil
import argparse
from pathlib import Path

BACKUP_DIR = Path(os.getenv("BACKUP_DIR", "/var/backups/warpsmith"))
DB_PATH = Path(os.getenv("DB_PATH", "simulator.db"))


def find_latest_backup() -> Path:
    """Найти самый свежий .sqlite.gz в BACKUP_DIR."""
    backups = sorted(BACKUP_DIR.glob("simulator-*.sqlite.gz"))
    if not backups:
        print(f"Ошибка: backup-файлы не найдены в {BACKUP_DIR}", file=sys.stderr)
        sys.exit(1)
    return backups[-1]


def find_backup_by_date(date_str: str) -> Path:
    """Найти backup по дате."""
    candidates = list(BACKUP_DIR.glob(f"simulator-{date_str}.sqlite.gz"))
    candidates += list(BACKUP_DIR.glob(f"simulator-weekly-{date_str}.sqlite.gz"))
    if not candidates:
        print(f"Ошибка: backup за {date_str} не найден", file=sys.stderr)
        sys.exit(1)
    return candidates[0]


def restore_from_file(gz_path: Path) -> None:
    """Восстановить базу из gz файла."""
    if not gz_path.exists():
        print(f"Ошибка: {gz_path} не найден", file=sys.stderr)
        sys.exit(1)

    # Распаковать
    temp_path = gz_path.with_suffix("")  # убрать .gz
    with gzip.open(gz_path, "rb") as f_in:
        with open(temp_path, "wb") as f_out:
            shutil.copyfileobj(f_in, f_out)

    # Подтверждение
    print(f"Будет восстановлен backup: {gz_path.name}")
    print(f"Целевой файл: {DB_PATH.absolute()}")
    confirm = input("Продолжить? (y/N): ")
    if confirm.lower() != "y":
        temp_path.unlink()
        print("Отменено")
        sys.exit(0)

    # Скопировать поверх существующей БД
    shutil.copy2(temp_path, DB_PATH)
    temp_path.unlink()

    print(f"✓ База восстановлена из {gz_path.name}")
    print(f"  Размер: {DB_PATH.stat().st_size / 1024:.1f} KB")


def main() -> None:
    parser = argparse.ArgumentParser(description="Восстановить SQLite из backup")
    parser.add_argument("--date", help="Дата backup (YYYY-MM-DD)")
    parser.add_argument("--file", help="Путь к .sqlite.gz файлу")
    args = parser.parse_args()

    if args.file:
        gz_path = Path(args.file)
    elif args.date:
        gz_path = find_backup_by_date(args.date)
    else:
        gz_path = find_latest_backup()

    restore_from_file(gz_path)


if __name__ == "__main__":
    main()
```

## backup-cron.sh

```bash
#!/usr/bin/env bash
# scripts/backup-cron.sh — запуск backup.py с логированием
set -euo pipefail

BACKUP_DIR="${BACKUP_DIR:-/var/backups/warpsmith}"
LOG_DIR="${LOG_DIR:-/var/log/warpsmith}"
TIMESTAMP=$(date +%Y-%m-%d-%H%M%S)

mkdir -p "$BACKUP_DIR" "$LOG_DIR"

python scripts/backup.py >> "$LOG_DIR/backup-$TIMESTAMP.log" 2>&1

# Очистить логи старше 30 дней
find "$LOG_DIR" -name "backup-*.log" -mtime +30 -delete
```

import os
import gzip
import shutil
import sqlite3
from pathlib import Path
from datetime import datetime


def test_backup_creates_file(tmp_path):
    """Проверка, что backup.py создаёт .sqlite.gz."""
    # Создать тестовую БД
    db_path = tmp_path / "test.db"
    conn = sqlite3.connect(str(db_path))
    conn.execute("CREATE TABLE test (id INT)")
    conn.execute("INSERT INTO test VALUES (1)")
    conn.commit()
    conn.close()

    backup_dir = tmp_path / "backups"

    # Запустить backup
    os.environ["DB_PATH"] = str(db_path)
    os.environ["BACKUP_DIR"] = str(backup_dir)
    from scripts.backup import backup_database

    gz_path = backup_database()

    assert gz_path.exists()
    assert gz_path.suffix == ".gz"
    assert gz_path.stat().st_size > 0


def test_restore_recovers_data(tmp_path):
    """Проверка, что restore.py восстанавливает данные."""
    # Создать БД, backup, удалить БД
    db_path = tmp_path / "test.db"
    conn = sqlite3.connect(str(db_path))
    conn.execute("CREATE TABLE users (name TEXT)")
    conn.execute("INSERT INTO users VALUES ('test_user')")
    conn.commit()
    conn.close()

    backup_dir = tmp_path / "backups"
    os.environ["DB_PATH"] = str(db_path)
    os.environ["BACKUP_DIR"] = str(backup_dir)

    from scripts.backup import backup_database
    gz_path = backup_database()

    # Удалить оригинал
    db_path.unlink()

    # Восстановить
    from scripts.restore import restore_from_file
    restore_from_file(gz_path)

    # Проверить данные
    conn = sqlite3.connect(str(db_path))
    cur = conn.execute("SELECT name FROM users")
    assert cur.fetchone()[0] == "test_user"
    conn.close()
