# F4.4: Synergy Hints — Leader Compatibility, Transport Capacity

**Файлы:** `web/static/synergy_hints.js` (создать),
`web/routes/api.py` (расширение — /api/rosters/synergies),
`web/templates/partials/synergy_panel.html` (создать)
**Зависимости:** F2.12 (Leader Compatibility), F2.11 (Team Builder UI),
F4.2 (Unit Modal — для данных о лидерах/транспорте)
**Часы:** 4h

**Цель (SMART):** Реализовать систему подсказок синергий в Team Builder: (1) подсветка
совместимости лидер-отряд, (2) транспортная вместимость, (3) мягкие рекомендации из wiki
synergies. Критерий: Alpine.js проверяет ростер на каждое изменение, подсвечивает зелёным
совместимые пары, жёлтым — предупреждения, красным — ошибки; минимум 3 типа подсказок
(leader, transport, synergy). 4 часа.

## Core Types

```python
from pydantic import BaseModel
from typing import Optional

class SynergyCheck(BaseModel):
    type: str                          # "leader" | "transport" | "synergy"
    severity: str                      # "info" | "warning" | "error"
    source_unit: str                   # имя юнита, инициировавшего проверку
    target_unit: Optional[str] = None  # имя связанного юнита (может быть None)
    message: str                       # человеко-читаемое сообщение
    icon: str = "💡"                   # иконка для отображения
    action_url: Optional[str] = None   # ссылка на действие (добавить юнит)


@router.post("/api/rosters/synergies")
async def check_roster_synergies(data: dict):
    """
    Проверить ростер на синергии.
    request: { "faction": str, "units": list[{"name": str, "squad_size": int, ...}] }
    response: { "checks": list[SynergyCheck], "score": int }
    """
    checks = []
    unit_objs = [registry.get_unit(u["name"]) for u in data["units"]]

    # 1. Leader → Bodyguard compatibility
    leaders = [u for u in unit_objs if getattr(u, "is_leader", False)]
    bodyguards = [u for u in unit_objs if not getattr(u, "is_leader", False)]

    for leader in leaders:
        compatible = []
        incompatible = []
        for bg in bodyguards:
            if _can_lead(leader, bg):
                compatible.append(bg.name)
            else:
                incompatible.append(bg.name)

        if compatible:
            checks.append(SynergyCheck(
                type="leader", severity="info",
                source_unit=leader.name,
                message=f"{leader.name} can lead: {', '.join(compatible)}",
                icon="✅",
            ))
        if incompatible and not compatible:
            checks.append(SynergyCheck(
                type="leader", severity="error",
                source_unit=leader.name,
                message=f"{leader.name} cannot lead any unit in this roster",
                icon="❌",
            ))

    # 2. Transport capacity
    transports = [u for u in unit_objs if getattr(u, "transport_capacity", None)]
    for transport in transports:
        capacity = transport.transport_capacity
        # Найти юнитов с флагом "can_embark"
        embarked_count = 0
        for u in data["units"]:
            if u["name"] != transport.name and _can_embark(u, transport):
                embarked_count += u.get("squad_size", 1)

        if embarked_count > capacity:
            checks.append(SynergyCheck(
                type="transport", severity="error",
                source_unit=transport.name,
                message=f"{transport.name} can carry {capacity} models, "
                        f"but roster has {embarked_count} eligible models",
                icon="🚫",
            ))
        elif embarked_count == 0:
            checks.append(SynergyCheck(
                type="transport", severity="info",
                source_unit=transport.name,
                message=f"{transport.name} is empty — no infantry to transport",
                icon="🚌",
            ))

    # 3. Wiki synergy hints (from YAML frontmatter)
    for unit in unit_objs:
        synergies = getattr(unit, "synergies", [])
        for syn in synergies:
            matched_targets = [u for u in data["units"]
                               if u["name"].lower() == syn.get("with", "").lower()]
            if matched_targets:
                checks.append(SynergyCheck(
                    type="synergy",
                    severity=syn.get("type", "info"),
                    source_unit=unit.name,
                    target_unit=syn["with"],
                    message=syn.get("text", ""),
                    icon="💡",
                ))

    # Score: count info/warning/error
    score = sum(
        (3 if c.severity == "error" else
         1 if c.severity == "warning" else 0)
        for c in checks
    )
    return {"checks": checks, "score": score}
```

## Alpine.js Controller

```javascript
// web/static/synergy_hints.js
function synergyHints() {
    return {
        synergyChecks: [],
        debounceTimer: null,
        lastCheck: '',

        get errorCount() {
            return this.synergyChecks.filter(c => c.severity === 'error').length;
        },
        get warningCount() {
            return this.synergyChecks.filter(c => c.severity === 'warning').length;
        },
        get infoCount() {
            return this.synergyChecks.filter(c => c.severity === 'info').length;
        },

        checkSynergies() {
            // Debounce to avoid excessive API calls
            clearTimeout(this.debounceTimer);
            this.debounceTimer = setTimeout(() => {
                const payload = {
                    faction: this.faction,
                    units: this.roster,
                };
                const payloadStr = JSON.stringify(payload);
                if (payloadStr === this.lastCheck) return; // No change
                this.lastCheck = payloadStr;

                if (this.roster.length === 0) {
                    this.synergyChecks = [];
                    return;
                }

                fetch('/api/rosters/synergies', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: payloadStr,
                })
                .then(r => r.json())
                .then(data => {
                    this.synergyChecks = data.checks || [];
                })
                .catch(() => {
                    this.synergyChecks = [];
                });
            }, 500); // 500ms debounce
        },
    };
}
```

```javascript
// Helper functions in synergyHints()
getSynergyDot(unitName) {
    const checks = this.synergyChecks.filter(c =>
        c.source_unit === unitName || c.target_unit === unitName);
    if (checks.some(c => c.severity === 'error')) return 'bg-red-500';
    if (checks.some(c => c.severity === 'warning')) return 'bg-yellow-500';
    if (checks.some(c => c.severity === 'info')) return 'bg-green-500';
    return 'bg-gray-600';
},
```

## Edge Cases

| Ситуация | Поведение |
|----------|-----------|
| Ростер пуст | Панель скрыта или показывает "Add units to check synergies" |
| Все синергии хорошие | Зелёный бейдж "X synergies", нет предупреждений |
| Лидер без отряда | Красная плашка, подсветка названия лидера |
| Транспорт переполнен | Красная плашка с подсчётом моделей |
| 500ms debounce | При быстром добавлении 3 юнитов — 1 запрос, а не 3 |
| Нет synergy данных в wiki | Панель показывает "No synergy data" |

```python
def test_synergies_leader_compatible(client):
    resp = client.post("/api/rosters/synergies", json={
        "faction": "orks",
        "units": [
            {"name": "Warboss", "squad_size": 1},
            {"name": "Boyz", "squad_size": 10},
        ],
    })
    assert resp.status_code == 200
    checks = resp.json()["checks"]
    leader_checks = [c for c in checks if c["type"] == "leader"]
    assert any("can lead" in c["message"] for c in leader_checks)

def test_synergies_transport_overcapacity(client):
    resp = client.post("/api/rosters/synergies", json={
        "faction": "orks",
        "units": [
            {"name": "Trukk", "squad_size": 1},
            {"name": "Boyz", "squad_size": 20},
        ],
    })
    checks = resp.json()["checks"]
    transport_checks = [c for c in checks if c["type"] == "transport"]
    assert len(transport_checks) > 0
    # Trukk capacity is 11, Boyz 20 exceeds it
    assert any("can carry" in c["message"] for c in transport_checks)

def test_synergies_empty_roster(client):
    resp = client.post("/api/rosters/synergies", json={
        "faction": "orks", "units": [],
    })
    assert resp.status_code == 200
    assert len(resp.json()["checks"]) == 0

def test_synergies_no_faction(client):
    resp = client.post("/api/rosters/synergies", json={
        "faction": "", "units": [],
    })
    assert resp.status_code == 200
```
