# F2.1: Game State Dataclass

**Файл:** `backend/state/game_state.py`
**Зависимости:** F1.1 (Unit), F1.9 (API)
**Часы:** 4h
**Обновлён:** 2026-05-04 (приведён в соответствие с реальной реализацией)

**Цель (SMART):** Создать GameState dataclass, хранящий позиции всех юнитов, раны, CP, VP, номер раунда и фазу. Критерий: можно создать состояние «2 армии на карте, 1 раунд, фаза Shooting», управлять юнитами (move, damage), переключать фазы. 4 часа.

## Data Model

```python
from dataclasses import dataclass, field
from enum import Enum
from typing import Optional
import numpy as np


class TerrainType(Enum):
    """Типы террейна на карте."""
    OPEN_GROUND = "open_ground"
    DIFFICULT_TERRAIN = "difficult_terrain"
    DANGEROUS_TERRAIN = "dangerous_terrain"
    IMPASSABLE = "impassable"


class GamePhase(Enum):
    """Фазы хода Warhammer 40k."""
    COMMAND = "command"
    MOVEMENT = "movement"
    SHOOTING = "shooting"
    CHARGE = "charge"
    FIGHT = "fight"
    MORALE = "morale"


@dataclass
class UnitState:
    """Состояние конкретного юнита/отряда на поле."""
    unit_id: str
    name: str
    faction: str
    position: tuple[int, int]          # (x, y) на карте (не отдельный Position dataclass)
    current_wounds: int                 # текущие раны
    max_wounds: int                     # макс. раны
    models_remaining: int               # сколько моделей живо
    models_total: int                   # всего моделей в отряде
    leadership: int
    objective_control: int
    is_warlord: bool = False
    has_moved: bool = False
    has_shot: bool = False
    has_charged: bool = False
    has_fought: bool = False
    is_pinned: bool = False
    command_points: int = 0
    is_engaged: bool = False            # в ближнем бою
    is_fighting: bool = False           # активировался в фазе Fight
    is_battle_shocked: bool = False     # battle-shock

    @property
    def is_above_half_strength(self) -> bool:
        return self.models_remaining > (self.models_total / 2)

    @property
    def is_alive(self) -> bool:
        return self.models_remaining > 0 and self.current_wounds > 0


@dataclass
class PlayerState:
    """Один игрок/армия. Заменяет RosterState из ранних спек."""
    player_id: str
    name: str
    faction: str
    command_points: int = 6
    victory_points: int = 0
    units: dict[str, UnitState] = field(default_factory=dict)
    stratagems_used: list[str] = field(default_factory=list)
    is_active: bool = True
    command_priority: bool = False       # приоритет хода в раунде

    @property
    def total_objective_control(self) -> int:
        return sum(u.objective_control for u in self.units.values() if u.is_alive)

    @property
    def warlord_unit(self) -> UnitState | None:
        for unit in self.units.values():
            if unit.is_warlord:
                return unit
        return None


@dataclass
class GameState:
    """Полное состояние игры."""
    game_id: str
    mission_name: str
    map_width: int = 6
    map_height: int = 4
    current_round: int = 1
    current_phase: GamePhase = GamePhase.COMMAND
    active_player: str | None = None
    players: dict[str, PlayerState] = field(default_factory=dict)   # ключ = player_id
    terrain_map: np.ndarray | None = None
    deployment_zones: dict[str, list[tuple[int, int]]] = field(default_factory=dict)
    objectives: dict[str, tuple[int, int]] = field(default_factory=dict)
    max_rounds: int = 5
    previous_round_priority_player_id: str | None = None
    victory_conditions: dict = field(default_factory=dict)
    game_log: list[str] = field(default_factory=list)
    mission: Optional["Mission"] = None
```

## Ключевые методы GameState

```python
# ── Управление фазами ────────────────────────────────────────────

def next_phase(self):
    """Переключить на следующую фазу. При переходе через Morale → Command
    начинается новый раунд, сброс флагов юнитов, определение command priority."""
    phases = list(GamePhase)
    current_index = phases.index(self.current_phase)
    next_index = (current_index + 1) % len(phases)

    if next_index == 0:  # Новый раунд
        self.current_round += 1
        for player in self.players.values():
            for unit in player.units.values():
                unit.has_moved = False
                unit.has_shot = False
                unit.has_charged = False
                unit.has_fought = False
                unit.is_engaged = False
                unit.is_fighting = False
                unit.is_battle_shocked = False
        self._determine_command_priority()

    self.current_phase = phases[next_index]
    self.game_log.append(f"Phase changed to {self.current_phase.value}")


def _determine_command_priority(self):
    """Раунд 1: случайный бросок. Раунд 2+: приоритет переходит к другому игроку."""


# ── Действия ─────────────────────────────────────────────────────

def move_unit(self, unit_id: str, new_position: tuple[int, int]) -> bool
def deal_damage(self, unit_id: str, damage: int) -> bool
def add_victory_points(self, player_id: str, points: int)

def get_unit_at_position(self, x: int, y: int) -> UnitState | None
def _is_valid_move(self, unit: UnitState, new_position: tuple[int, int]) -> bool


# ── Свойства ─────────────────────────────────────────────────────

@property
def is_game_over(self) -> bool          # раунд > max_rounds или VP >= 10

@property
def winner(self) -> str | None          # игрок с макс VP (None при ничьей)


# ── Сводка ────────────────────────────────────────────────────────

def get_game_summary(self) -> dict:
    """Вернуть JSON-совместимую сводку: game_id, mission, round, phase,
    по каждому игроку: name, faction, cp, vp, units_alive, total_oc."""


# ── Фабрики ──────────────────────────────────────────────────────

def create_empty_game(game_id: str, mission_name: str = "Standard Mission") -> GameState
def create_standard_map() -> np.ndarray
```

## Отличия от ранней версии спеки (F2.1 → актуальный код)

| Аспект | Ранняя спека | Актуальный код |
|--------|-------------|----------------|
| `Position` | Отдельный `@dataclass Position(x, y)` | `tuple[int, int]` (проще, сериализуется нативно) |
| `RosterState` | `player_id, faction, detachment, units, total_pts, cp` | Заменён на `PlayerState` с полями `player_id, name, faction, command_points, victory_points, units` |
| `GameState.players` | `roster_a: RosterState, roster_b: RosterState` | `players: dict[str, PlayerState]` (поддерживает >2 игроков, lookup по id) |
| `next_phase()` | Отдельная функция | Метод `GameState.next_phase()` |
| `TerrainType` / `GamePhase` | Не указаны | Реальные `Enum` с 4 типами террейна и 6 фазами |
| Флаги юнита | `is_advancing, is_shooting, is_charging, cp_generated_this_turn` | `has_moved, has_shot, has_charged, has_fought` (проще, symlink с правилами 10th ed) |
| Методы | ❌ нет | `move_unit, deal_damage, add_victory_points, get_unit_at_position, get_game_summary, _determine_command_priority` |

---

## Edge Cases

| Ситуация | Поведение |
|----------|-----------|
| Переход через Morale → Command | Сброс флагов (has_moved, has_shot...), определение command priority |
| Юнит на Impassable terrain | `_is_valid_move` возвращает False |
| Все юниты мертвы | `is_game_over = True`, `winner` = игрок с живыми юнитами |
| Урон > current_wounds | `current_wounds` не уходит в минус (max(0, ...)) |
| Ничья по VP | `winner = None` |
| Раунд 1 command priority | Случайный выбор |
| Раунд 2+ command priority | Переходит к другому игроку |

```python
def test_game_state_creation():
    state = create_empty_game("game_1")
    assert state.current_round == 1
    assert state.current_phase == GamePhase.COMMAND
    assert len(state.players) == 0


def test_add_player_with_units():
    state = create_empty_game("game_1")
    state.players["1"] = PlayerState(
        player_id="1", name="Player 1", faction="orks",
        units={"warboss": UnitState(
            unit_id="warboss", name="Warboss", faction="orks",
            position=(1, 1), current_wounds=5, max_wounds=5,
            models_remaining=1, models_total=1,
            leadership=8, objective_control=3, is_warlord=True,
        )}
    )
    assert state.players["1"].warlord_unit is not None
    assert state.players["1"].total_objective_control == 3


def test_phase_transition():
    state = create_empty_game("game_1")
    state.current_phase = GamePhase.COMMAND
    state.next_phase()
    assert state.current_phase == GamePhase.MOVEMENT


def test_new_round_flags_reset():
    state = create_empty_game("game_1")
    state.players["1"] = PlayerState(
        player_id="1", name="P1", faction="orks",
        units={"test": UnitState(
            unit_id="test", name="Test", faction="orks",
            position=(0, 0), current_wounds=1, max_wounds=1,
            models_remaining=1, models_total=1,
            leadership=7, objective_control=1, has_shot=True,
        )}
    )
    # Пройти все фазы до конца (COMMAND → ... → MORALE → новый раунд)
    for _ in range(6):
        state.next_phase()
    assert state.current_round == 2
    assert state.players["1"].units["test"].has_shot is False


def test_serialization_roundtrip():
    state = create_empty_game("game_1")
    # dataclasses.asdict + json.dumps — нативная сериализация
    import json
    from dataclasses import asdict
    data = json.dumps(asdict(state), default=str)
    assert "game_id" in data
    assert "mission_name" in data
```
