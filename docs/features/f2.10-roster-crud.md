# F2.10: Roster CRUD — SQLite Save/Load/Delete

**Файл:** `web/routes/api_rosters.py` (расширение), `backend/db/database.py` (расширение)
**Зависимости:** F2.9 (Roster Validation)
**Часы:** 2h

**Цель (SMART):** Реализовать CRUD для ростера через SQLite: создание, чтение списка/по id, обновление, удаление. Критерий: POST /api/rosters создаёт запись, GET /api/rosters возвращает список, DELETE /api/rosters/{id} удаляет; максимальный лимит ростера для Free tier — 1. 2 часа.


## API Endpoints

```python
from pydantic import BaseModel
from typing import Optional

class RosterCreate(BaseModel):
    name: str
    faction: str
    pts_limit: int = 2000
    detachment: Optional[str] = None
    units: list[dict]           # [{"unit_name": "Boyz", "squad_size": 10}, ...]
    is_public: bool = False

class RosterUpdate(BaseModel):
    name: Optional[str] = None
    units: Optional[list[dict]] = None
    is_public: Optional[bool] = None


@router.post("/api/rosters")
async def create_roster(
    data: RosterCreate,
    user = Depends(get_current_user),
):
    """Создать новый ростер."""

    # Check Free tier limit
    user_features = UserFeatures.for_user(user)
    current_count = db.fetchone(
        "SELECT COUNT(*) as cnt FROM rosters WHERE user_id = ?", (user.id,)
    )["cnt"]
    if current_count >= user_features.max_rosters:
        raise HTTPException(403, detail="Max rosters limit reached. Upgrade to Premium.")

    # Validate
    units_list = [(u["unit_name"], u["squad_size"]) for u in data.units]
    validation = validate_roster(units_list, registry.load())
    if not validation.is_valid:
        raise HTTPException(400, detail={
            "error": "roster_invalid",
            "validation_errors": [e.__dict__ for e in validation.errors],
        })

    # Create
    cur = db.execute(
        """INSERT INTO rosters (user_id, name, faction, pts_limit, detachment, units, is_public)
           VALUES (?, ?, ?, ?, ?, ?, ?)""",
        (user.id, data.name, data.faction, data.pts_limit,
         data.detachment, json.dumps(data.units), int(data.is_public)),
    )
    db.commit()
    return {"id": cur.lastrowid, **data.model_dump()}


@router.get("/api/rosters")
async def list_rosters(
    user = Depends(get_current_user),
    public_only: bool = False,
):
    """Список ростереров текущего пользователя."""

    if public_only:
        rows = db.fetchall(
            "SELECT * FROM rosters WHERE is_public = 1 ORDER BY updated_at DESC"
        )
    else:
        rows = db.fetchall(
            "SELECT * FROM rosters WHERE user_id = ? ORDER BY updated_at DESC",
            (user.id,),
        )
    return {"rosters": [dict(r) for r in rows]}


@router.get("/api/rosters/{roster_id}")
async def get_roster(roster_id: int, user = Depends(get_current_user)):
    """Получить ростер по id."""

    row = db.fetchone("SELECT * FROM rosters WHERE id = ?", (roster_id,))
    if not row:
        raise HTTPException(404, detail="Roster not found")
    if row["user_id"] != user.id and not row["is_public"]:
        raise HTTPException(403, detail="Not authorized")

    return dict(row)


@router.delete("/api/rosters/{roster_id}")
async def delete_roster(roster_id: int, user = Depends(get_current_user)):
    """Удалить ростер."""

    row = db.fetchone("SELECT user_id FROM rosters WHERE id = ?", (roster_id,))
    if not row:
        raise HTTPException(404, detail="Roster not found")
    if row["user_id"] != user.id:
        raise HTTPException(403, detail="Not authorized")

    db.execute("DELETE FROM rosters WHERE id = ?", (roster_id,))
    db.commit()
    return {"status": "deleted", "id": roster_id}
```

```python
def test_create_roster(client, auth_headers):
    resp = client.post("/api/rosters", json={
        "name": "My Ork Horde",

def test_free_tier_limit(client, auth_headers):
    """Free tier: max 1 roster."""
    client.post("/api/rosters", json={"name": "R1", ...}, headers=auth_headers)
    resp = client.post("/api/rosters", json={"name": "R2", ...}, headers=auth_headers)
    assert resp.status_code == 403  # limit reached

def test_delete_roster(client, auth_headers):
    create = client.post("/api/rosters", json={...}, headers=auth_headers)
    roster_id = create.json()["id"]
    resp = client.delete(f"/api/rosters/{roster_id}", headers=auth_headers)
    assert resp.status_code == 200
```
