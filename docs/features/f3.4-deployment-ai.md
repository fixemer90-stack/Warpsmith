# F3.4: Deployment AI — Zone Placement Logic

**Файл:** `backend/engine/ai/deployment.py` (создать)
**Зависимости:** F2.2 (2D Map), F2.4 (Missions), F2.1 (Game State)
**Часы:** 3h

**Цель (SMART):** Реализовать DeploymentAI, размещающий юниты в deploy zone согласно миссии, с поддержкой: стандартной расстановки (12" от центра), Hammer & Anvil (длинные стороны), Search & Destroy (углы), Dawn of War (полоски). Критерий: все 4 deployment типа размещают юниты в корректных deploy zone (минимум 8 тестов), юниты не накладываются друг на друга и не выходят за границы карты. 3 часа.

## Deployment Types

```python
from enum import Enum
from dataclasses import dataclass
from typing import Optional


class DeploymentType(Enum):
    STANDARD = "standard"             # 12" от центра
    HAMMER_AND_ANVIL = "hammer_and_anvil"  # длинные стороны
    SEARCH_AND_DESTROY = "search_and_destroy"  # углы
    DAWN_OF_WAR = "dawn_of_war"       # полоски


@dataclass
class DeploymentZone:
    """Прямоугольная зона развёртывания."""
    player: int
    x_min: int
    x_max: int
    y_min: int
    y_max: int
    label: str = ""
```

## Zone Calculation

```python
def get_deployment_zone(deployment_type: DeploymentType,
                        player: int,
                        map_size: tuple[int, int]) -> DeploymentZone:
    """Вернуть deploy zone для игрока на карте map_size (width, height).

    STANDARD:
      P1: x ∈ [0, width/3], P2: x ∈ [2*width/3, width]

    HAMMER_AND_ANVIL:
      P1: x ∈ [0, width/4], P2: x ∈ [3*width/4, width]

    SEARCH_AND_DESTROY:
      P1: угол (0, 0) — (width/3, height/3)
      P2: угол (2*width/3, 2*height/3) — (width, height)

    DAWN_OF_WAR:
      P1: x ∈ [0, width/2], y ∈ [0, height/4]
      P2: x ∈ [0, width/2], y ∈ [3*height/4, height]
    """
    w, h = map_size

    if deployment_type == DeploymentType.STANDARD:
        if player == 1:
            return DeploymentZone(1, 0, w // 3, 0, h, "P1 Standard")
        else:
            return DeploymentZone(2, 2 * w // 3, w, 0, h, "P2 Standard")

    elif deployment_type == DeploymentType.HAMMER_AND_ANVIL:
        if player == 1:
            return DeploymentZone(1, 0, w // 4, 0, h, "P1 Hammer")
        else:
            return DeploymentZone(2, 3 * w // 4, w, 0, h, "P2 Anvil")

    elif deployment_type == DeploymentType.SEARCH_AND_DESTROY:
        if player == 1:
            return DeploymentZone(1, 0, w // 3, 0, h // 3, "P1 S&D")
        else:
            return DeploymentZone(2, 2 * w // 3, w, 2 * h // 3, h, "P2 S&D")

    elif deployment_type == DeploymentType.DAWN_OF_WAR:
        if player == 1:
            return DeploymentZone(1, 0, w // 2, 0, h // 4, "P1 DoW")
        else:
            return DeploymentZone(2, 0, w // 2, 3 * h // 4, h, "P2 DoW")

    raise ValueError(f"Unknown deployment type: {deployment_type}")
```

## Placement Strategy

```python
@dataclass
class Placement:
    """Результат размещения одного юнита."""
    unit_id: str
    x: int
    y: int
    is_in_cover: bool = False
    is_on_objective: bool = False


def place_units(roster: RosterState,
                deployment_type: DeploymentType,
                map_: GameMap,
                objectives: list[Objective]) -> list[Placement]:
    """Разместить все юниты ростера в deploy zone.

    Стратегия размещения:
    1. Warlord → центр зоны (лидерский бонус)
    2. Melee units → передний край deploy zone
    3. Ranged units → задний край / в cover
    4. Epic Heroes → рядом с подходящим отрядом
    5. Транспорты → вместить отряды, поставить в cover
    """
    zone = get_deployment_zone(deployment_type, roster.player_id,
                               (map_.width, map_.height))
    placements = []

    # Сортировать юниты: Warlord → Melee → Ranged → Support
    sorted_units = _sort_units_by_role(roster)

    occupied = set()
    for unit in sorted_units:
        pos = _find_best_position(unit, zone, map_, objectives, occupied)
        if pos:
            placements.append(Placement(
                unit_id=unit.unit_id,
                x=pos[0], y=pos[1],
                is_in_cover=_has_cover_at(map_, pos[0], pos[1]),
                is_on_objective=_is_on_objective(pos, objectives),
            ))
            occupied.add(pos)

    return placements


def _sort_units_by_role(roster: RosterState) -> list[UnitState]:
    """Сортировать: Warlord → Transport → Melee → Battleline → Ranged → Support."""
    def sort_key(unit):
        if unit.unit_id == roster.warlord_id:
            return 0
        if 'transport' in unit.keywords:
            return 1
        if _is_melee_unit(unit):
            return 2
        if 'battleline' in unit.keywords:
            return 3
        if _is_ranged_unit(unit):
            return 4
        return 5

    return sorted(roster.units, key=sort_key)


def _find_best_position(unit: UnitState, zone: DeploymentZone,
                        map_: GameMap, objectives: list[Objective],
                        occupied: set[tuple[int, int]]) -> Optional[tuple[int, int]]:
    """Найти лучшую позицию для юнита в зоне.

    - Не занято другим юнитом (margin 2 клетки)
    - Приоритет cover (если ranged)
    - Приоритет objective (если melee с высоким OC)
    - Не ближе 3" от другого юнита (coherency)
    """
    best_score = -1
    best_pos = None

    for x in range(zone.x_min, zone.x_max):
        for y in range(zone.y_min, zone.y_max):
            if _is_occupied(x, y, occupied, margin=2):
                continue
            if not _is_walkable(map_, x, y):
                continue

            score = 0.0
            # Cover бонус для ranged
            if _is_ranged_unit(unit) and _has_cover_at(map_, x, y):
                score += 0.5
            # Objective бонус для melee
            if _is_melee_unit(unit):
                for obj in objectives:
                    if _distance((x, y), (obj.x, obj.y)) <= 3:
                        score += 0.8
            # Центр зоны (warlord bonus)
            if unit.unit_id == unit.roster.warlord_id:  # noqa
                center = ((zone.x_min + zone.x_max) // 2,
                          (zone.y_min + zone.y_max) // 2)
                score += 0.3 * (1 - _distance((x, y), center) / 20.0)

            if score > best_score:
                best_score = score
                best_pos = (x, y)

    return best_pos or _find_first_free(zone, occupied)
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Deploy zone полностью занята террейном | Использовать ближайшую свободную клетку |
| Больше юнитов чем свободных клеток | Разместить вплотную (min coherency 1") |
| Юнит не влезает в deploy zone (большая база) | Центрировать, проверить все 4 угла базы |
| Нет objectives на карте | Разместить равномерно по deploy zone |
| Транспорт с embarked отрядами | Разместить транспорт, отряды внутри |
| Warlord уничтожен до размещения | Пропустить warlord sorting |
| Тип deployment неизвестен | По умолчанию STANDARD |
| Карта нестандартного размера | Зоны пропорциональны размеру карты |
