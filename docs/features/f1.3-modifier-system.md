# F1.3: Modifier System
**Цель (SMART):** Реализовать Modifier dataclass, apply_modifiers() с ±1 cap (10ed rule), reroll, sustained_hits, auto_hit. Критерий: 10 юнит-тестов покрывают cancel (+1/-1 → 0), cap (+2 → +1), sustained_hits=1, lethal_hits, auto_hit (Torrent). 3 часа.

**Файл:** `backend/engine/modifiers.py` (создать)
**Зависимости:** F1.5 (Dice Pool)

## Data Model

```python
from dataclasses import dataclass, field
from typing import Any, Optional, Literal

Operation = Literal[
    "add", "subtract",
    "reroll_hits", "reroll_wounds", "reroll_saves",
    "reroll_ones_hit", "reroll_ones_wound",
    "sustained_hits", "lethal_hits", "devastating_wounds",
    "ignore_cover", "ignore_modifiers",
    "auto_hit", "auto_wound",
]

TargetStep = Literal[
    "attack_count", "hit_roll", "wound_roll",
    "save_roll", "damage", "fnp_roll",
]

@dataclass
class Modifier:
    target: TargetStep
    operation: Operation
    value: Any = 0                         # int для sustained = число доп.хитов
    source: str = "ability"                # ability / wargear / stratagem / detachment
    condition: Optional[dict] = None       # {"half_range": True, "stationary": True}

@dataclass
class ModifierContext:
    """Контекст, в котором применяются модификаторы."""
    attacker: Any                          # Unit
    defender: Any                          # Unit
    weapon: Any                            # Weapon
    distance: Optional[int] = None         # дистанция в дюймах
    is_stationary: bool = False            # двигалась ли модель
    squad_size: int = 1                    # размер отряда (для Blast)

@dataclass
class ModifierResult:
    """Результат применения модификаторов к шагу."""
    target_value: int                      # модифицированный target
    extra_rolls: int = 0                   # доп. броски (sustained hits)
    auto_success: bool = False             # lethal hits / torrent
    ignore_save: bool = False              # devastating wounds
    reroll: bool = False                   # twin-linked
```

## Core API

```python
def apply_modifiers(
    step: TargetStep,
    base_target: int,                      # базовый target (4 для 4+)
    modifiers: list[Modifier],
    context: ModifierContext,
    rng: np.random.Generator,
) -> ModifierResult:
    """
    Применить все релевантные модификаторы к step.

    10ed rule: hit/wound roll modifiers capped at ±1.
    Если есть и +1 и -1 → net 0.
    Natural 1 always fails, natural 6 always succeeds.
    """
    total_delta = 0
    result = ModifierResult(target_value=base_target)

    for mod in modifiers:
        if mod.target != step:
            continue
        if mod.condition and not _check_condition(mod.condition, context):
            continue

        if mod.operation == "add":
            total_delta += mod.value
        elif mod.operation == "subtract":
            total_delta -= mod.value
        elif mod.operation == "sustained_hits":
            result.extra_rolls = mod.value   # +1/+2/+3 hits on crit
        elif mod.operation == "lethal_hits":
            result.auto_success = True       # hit=6 → auto wound
        elif mod.operation == "devastating_wounds":
            result.ignore_save = True        # wound=6 → ignore save
        elif mod.operation == "auto_hit":
            result.auto_success = True       # Torrent
        # ... etc

    # Cap hit/wound modifiers at ±1 (10ed rule)
    if step in ("hit_roll", "wound_roll"):
        total_delta = max(-1, min(1, total_delta))

    result.target_value = base_target + total_delta
    result.target_value = max(2, min(6, result.target_value))

    return result


def _check_condition(cond: dict, ctx: ModifierContext) -> bool:
    """Проверить условия применения модификатора."""
    if cond.get("half_range") and ctx.distance:
        if ctx.weapon and ctx.weapon.range_max:
            return ctx.distance <= ctx.weapon.range_max / 2
    if cond.get("stationary") and not ctx.is_stationary:
        return False
    if cond.get("squad_size_min"):
        return ctx.squad_size >= cond["squad_size_min"]
    return True


def build_weapon_modifiers(weapon: Any) -> list[Modifier]:
    """Собрать модификаторы из tags оружия."""
    mods = []
    for tag in weapon.tags:
        if tag == "torrent":
            mods.append(Modifier("hit_roll", "auto_hit", source="wargear"))
        elif tag == "heavy":
            mods.append(Modifier("hit_roll", "add", 1, source="wargear",
                                  condition={"stationary": True}))
        elif tag == "twin_linked":
            mods.append(Modifier("wound_roll", "reroll_wounds", source="wargear"))
    return mods
```

## 10ed Rule: Reroll Logic

```python
def should_reroll(
    roll: int,
    target: int,
    modifiers: list[Modifier],
    step: TargetStep,
) -> bool:
    """Определить, нужно ли перебросить куб."""
    for mod in modifiers:
        if mod.target != step:
            continue
        if mod.operation == "reroll_all":
            return True
        if mod.operation == "reroll_ones" and roll == 1:
            return True
        if mod.operation == "reroll_hits" and step == "hit_roll":
            return True
        if mod.operation == "reroll_wounds" and step == "wound_roll":
            return True
    return False
```

## Edge Cases

| Ситуация | Поведение |
|----------|-----------|
| +1 to hit и -1 to hit | net 0 (cancel) |
| +2 to hit | capped at +1 |
| Sustained Hits + Lethal Hits | hit=6: extra hit + auto wound |
| Natural 1 to hit | always miss regardless of modifiers |
| Natural 6 to wound | always wound regardless of toughness |
| Twin-linked + reroll 1s | не стакаются — один reroll |

## Тесты

```python
def test_cap_modifier():
    mods = [Modifier("hit_roll", "add", 1), Modifier("hit_roll", "add", 1)]
    ctx = ModifierContext(...)
    result = apply_modifiers("hit_roll", 4, mods, ctx, ...)
    assert result.target_value == 3  # capped at +1, not +2

def test_cancel_modifier():
    mods = [Modifier("hit_roll", "add", 1), Modifier("hit_roll", "subtract", 1)]
    ctx = ModifierContext(...)
    result = apply_modifiers("hit_roll", 4, mods, ctx, ...)
    assert result.target_value == 4  # cancel out

def test_sustained_hits():
    mods = [Modifier("hit_roll", "sustained_hits", 1)]
    ctx = ModifierContext(...)
    result = apply_modifiers("hit_roll", 4, mods, ctx, ...)
    assert result.extra_rolls == 1
```
