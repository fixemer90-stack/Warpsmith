## Test: Natural 1 & 6

```python
def test_natural_one_always_fails():
    """Natural 1 на hit → всегда промах."""
    # Используем DicePool с фиксированным seed
    # и оружие с BS=2+
    pool = DicePool(seed=42)
    bolter = Weapon(name="Bolter", ..., attacks_dice=(0,0,1), skill=2, ...)
    marine = Unit(name="Tactical Marine", toughness=4, save=3, wounds=2, ...)

    # Monte Carlo: BS2+ means only 1/6 chance to miss (natural 1)
    result = simulate_weapon_attack(bolter, None, marine, pool=pool)
    # P(hit) should be ~5/6, not 1.0 (which would ignore natural 1)
    assert abs(result.stats.mean / result.avg_attacks - 5/6) < 0.02
```

## Test: DicePool

```python
def test_dice_reproducibility():
    pool1 = DicePool(seed=12345)
    pool2 = DicePool(seed=12345)
    assert np.array_equal(pool1.d6(100), pool2.d6(100))

def test_d3_range():
    pool = DicePool(seed=42)
    rolls = pool.d3(1000)
    assert set(rolls).issubset({1, 2, 3})

def test_simulate_batched_same_as_regular():
    pool = DicePool(seed=42)
    def one_damage(rng):
        return int(rng.integers(1, 3))
    a = pool.simulate(one_damage, n=1000)
    b = pool.simulate_batched(one_damage, n=1000, batch_size=100)
    np.testing.assert_array_equal(a, b)
```
