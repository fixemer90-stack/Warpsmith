# F2.4: Missions — Objectives, Scoring, Deployment Rules

**Файл:** `backend/state/mission.py` (создать)
**Зависимости:** F2.1 (Game State), F2.2 (Map)
**Часы:** 3h

**Цель (SMART):** Реализовать систему миссий с правилами высадки, условиями захвата точек и подсчётом VP каждую фазу Command. Критерий: 3 готовые миссии (Only War, Purge the Foe, Take and Hold), каждая задаёт deploy зоны, 3-5 objectives, условия победы на 5 раундов. 3 часа.

## Data Model

```python
from dataclasses import dataclass, field
from enum import Enum
from typing import Callable

class DeploymentType(Enum):
    DAWN_OF_WAR = "dawn_of_war"         # 24" deploy zone, 12" gap
    SEARCH_AND_DESTROY = "search_and_destroy"
    CRUCIBLE_OF_BATTLE = "crucible_of_battle"

class MissionObjective:
    """Точка захвата на карте."""
    def __init__(self, x: int, y: int, label: str = ""):
        self.x = x
        self.y = y
        self.label = label
        self.controlled_by: Optional[int] = None  # player_id
        self.is_contested: bool = False

@dataclass
class MissionConfig:
    """Конфигурация миссии."""
    name: str
    deployment: DeploymentType
    description: str
    objectives: list[MissionObjective]
    max_rounds: int = 5
    scoring_rule: str = "standard"  # standard, progressive, only_war
    deployment_zone_rows: int = 24  # 24" от края
    no_mans_land_start: int = 24
    no_mans_land_end: int = 36

@dataclass
class Mission:
    """Миссия в игре."""
    config: MissionConfig
    state: GameState

    def score_vp(self, player: int) -> int:
        """Подсчёт VP для игрока на текущей фазе Command."""
        vp = 0
        for obj in self.config.objectives:
            if obj.controlled_by == player and not obj.is_contested:
                vp += 1
        return vp
```

## Scoring Rules

```python
def score_standard(mission: Mission) -> dict[str, int]:
    """Standard scoring: VP = number of objectives controlled."""
    mission.update_objective_control()

    vp = {player_id: 0 for player_id in mission.state.players.keys()}
    for player_id in mission.state.players.keys():
        for obj in mission.config.objectives:
            if obj.controlled_by == player_id and not obj.is_contested:
                vp[player_id] += 1  # 1 VP per objective controlled
    return vp

def score_progressive(mission: Mission) -> dict[str, int]:
    """Progressive scoring: VP = objectives controlled + bonus for controlling more than opponent."""
    mission.update_objective_control()

    vp = {player_id: 0 for player_id in mission.state.players.keys()}
    player_ids = list(mission.state.players.keys())

    if len(player_ids) >= 2:
        p1_obj = 0
        p2_obj = 0

        for obj in mission.config.objectives:
            if obj.controlled_by == player_ids[0] and not obj.is_contested:
                p1_obj += 1
            elif obj.controlled_by == player_ids[1] and not obj.is_contested:
                p2_obj += 1

        vp[player_ids[0]] = p1_obj
        vp[player_ids[1]] = p2_obj

        # Bonus for controlling more objectives
        if p1_obj > p2_obj:
            vp[player_ids[0]] += 2
        elif p2_obj > p1_obj:
            vp[player_ids[1]] += 2

    return vp

def score_kill_points(mission: Mission) -> dict[str, int]:
    """VP = % of opponent's army destroyed."""
    vp = {pid: 0 for pid in mission.state.players.keys()}
    player_ids = list(mission.state.players.keys())

    if len(player_ids) >= 2:
        p1_killed = 0  # killed by p1 (p2's units)
        p2_killed = 0  # killed by p2 (p1's units)

        # Points destroyed by player 1 (against player 2)
        for unit in mission.state.players[player_ids[1]].units.values():
            if not unit.is_alive:
                p1_killed += unit.max_wounds * 10  # rough approximation

        # Points destroyed by player 2 (against player 1)
        for unit in mission.state.players[player_ids[0]].units.values():
            if not unit.is_alive:
                p2_killed += unit.max_wounds * 10

        p1_total = sum(u.max_wounds * 10 for u in mission.state.players[player_ids[0]].units.values())
        p2_total = sum(u.max_wounds * 10 for u in mission.state.players[player_ids[1]].units.values())

        if p2_total > 0:
            vp[player_ids[0]] = int((p1_killed / p2_total) * 100)
        if p1_total > 0:
            vp[player_ids[1]] = int((p2_killed / p1_total) * 100)

    return vp
```
