# F2.4: Missions — Objectives, Scoring, Deployment Rules

**Файл:** `backend/state/mission.py` (создать)
**Зависимости:** F2.1 (Game State), F2.2 (Map)
**Часы:** 3h

**Цель (SMART):** Реализовать систему миссий с правилами высадки, условиями захвата точек и подсчётом VP каждую фазу Command. Критерий: 3 готовые миссии (Only War, Purge the Foe, Take and Hold), каждая задаёт deploy зоны, 3-5 objectives, условия победы на 5 раундов. 3 часа.

## Data Model

```python
from dataclasses import dataclass, field
from enum import Enum
from typing import Callable

class DeploymentType(Enum):
    DAWN_OF_WAR = "dawn_of_war"         # 24" deploy zone, 12" gap
    SEARCH_AND_DESTROY = "search_and_destroy"
    CRUCIBLE_OF_BATTLE = "crucible_of_battle"

class MissionObjective:
    """Точка захвата на карте."""
    def __init__(self, x: int, y: int, label: str = ""):
        self.x = x
        self.y = y
        self.label = label
        self.controlled_by: Optional[int] = None  # player_id
        self.is_contested: bool = False

@dataclass
class MissionConfig:
    """Конфигурация миссии."""
    name: str
    deployment: DeploymentType
    description: str
    objectives: list[MissionObjective]
    max_rounds: int = 5
    scoring_rule: str = "standard"  # standard, progressive, only_war
    deployment_zone_rows: int = 24  # 24" от края
    no_mans_land_start: int = 24
    no_mans_land_end: int = 36

@dataclass
class Mission:
    """Миссия в игре."""
    config: MissionConfig
    state: GameState

    def score_vp(self, player: int) -> int:
        """Подсчёт VP для игрока на текущей фазе Command."""
        vp = 0
        for obj in self.config.objectives:
            if obj.controlled_by == player and not obj.is_contested:
                vp += 1
        return vp
```

## Scoring Rules

```python
def score_standard(mission: Mission) -> dict[int, int]:
    """VP = количество objectives под контролем на end of Command."""
    vp = {1: 0, 2: 0}
    for obj in mission.config.objectives:
        if obj.controlled_by and not obj.is_contested:
            vp[obj.controlled_by] += 5  # 5 VP per objective
    return vp

def score_progressive(mission: Mission) -> dict[int, int]:
    """VP = 5 per objective + 10 if hold more than opponent."""
    vp = {1: 0, 2: 0}
    for obj in mission.config.objectives:
        if obj.controlled_by and not obj.is_contested:
            vp[obj.controlled_by] += 5

    p1, p2 = vp[1], vp[2]
    if p1 > p2:
        vp[1] += 10
    elif p2 > p1:
        vp[2] += 10
    return vp

def score_kill_points(mission: Mission) -> dict[int, int]:
    """VP = % of opponent's army destroyed."""
    vp = {1: 0, 2: 0}
    for player in [1, 2]:
        roster = mission.state.roster_a if player == 1 else mission.state.roster_b
        opponent = mission.state.roster_b if player == 1 else mission.state.roster_a
        total_pts = opponent.total_pts
        killed_pts = _calc_killed_pts(opponent, mission.state)
        if total_pts > 0:
            vp[player] = int((killed_pts / total_pts) * 45)
    return vp
```
