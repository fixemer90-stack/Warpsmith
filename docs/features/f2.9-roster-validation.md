**Цель (SMART):** Реализовать валидацию ростера по правилам 10ed: PTS limit (2000), ровно 1 Warlord, максимум 3 копии одного юнита (6 для Battleline), уникальность Epic Heroes. Критерий: валидный ростер проходит, невалидный возвращает список ошибок с указанием конкретного нарушения. 3 часа.

## Validation Rules

```python
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class RosterValidationError:
    code: str                           # "pts_exceeded", "no_warlord", "too_many_copies"
    message: str                        # Человекочитаемое описание
    detail: Optional[dict] = None       # unit_name, current_count, max_count

@dataclass
class RosterValidationResult:
    is_valid: bool
    errors: list[RosterValidationError] = field(default_factory=list)
    total_pts: int = 0
    total_models: int = 0

    def add_error(self, code: str, message: str, **detail) -> None:
        self.errors.append(RosterValidationError(code, message, detail or None))
        self.is_valid = False
```

## Validator

```python
def validate_roster(
    units: list[tuple[str, int]],  # [(unit_name, squad_size), ...]
    unit_registry: dict[str, Unit],
    pts_limit: int = 2000,
) -> RosterValidationResult:
    """Валидация ростера по правилам 10ed."""
    result = RosterValidationResult()
    counts: dict[str, int] = {}   # unit_name -> count
    has_warlord = False
    total_pts = 0
    total_models = 0
    epic_heroes_seen: set[str] = set()

    for unit_name, squad_size in units:
        unit = unit_registry.get(unit_name)
        if unit is None:
            result.add_error("unknown_unit", f"Unit '{unit_name}' not found in registry",
                             unit_name=unit_name)
            continue

        # Count copies
        counts[unit_name] = counts.get(unit_name, 0) + 1

        # 3x cap (6x for Battleline)
        max_copies = 6 if "Battleline" in unit.keywords else 3
        if counts[unit_name] > max_copies:
            result.add_error("too_many_copies",
                f"Max {max_copies} copies of {unit_name} "
                f"({'Battleline' if 'Battleline' in unit.keywords else 'non-Battleline'})",
                unit_name=unit_name, max_count=max_copies, current_count=counts[unit_name])

        # Epic Hero: unique
        if unit.is_epic_hero:
            if unit_name in epic_heroes_seen:
                result.add_error("duplicate_epic_hero",
                    f"Epic Hero '{unit_name}' can only be taken once",
                    unit_name=unit_name)
            epic_heroes_seen.add(unit_name)

        # Warlord tracking
        if unit.can_be_warlord:
            has_warlord = True

        # Points
        unit_pts = unit.points * squad_size
        total_pts += unit_pts
        total_models += squad_size * unit.model_count[1]  # max per squad

    # PTS limit
    if total_pts > pts_limit:
        result.add_error("pts_exceeded",
            f"Total {total_pts}pts exceeds limit of {pts_limit}pts",
            pts_limit=pts_limit, total_pts=total_pts)

    # Warlord requirement
    if not has_warlord:
        result.add_error("no_warlord",
            "Roster must include at least one model that can be Warlord")

    # Empty roster
    if not units:
        result.add_error("empty_roster", "Roster is empty")

    result.total_pts = total_pts
    result.total_models = total_models
    return result
```

## Squad Size Validation

```python
def validate_squad_size(
    unit_name: str,
    squad_size: int,
    unit: Unit,
) -> Optional[RosterValidationError]:
    """Проверить размер отряда."""
    min_size, max_size = unit.model_count
    if squad_size < min_size:
        return RosterValidationError(
            "squad_too_small",
            f"{unit_name}: min {min_size} models, got {squad_size}",
            min_size=min_size, current_size=squad_size,
        )
    if squad_size > max_size:
        return RosterValidationError(
            "squad_too_large",
            f"{unit_name}: max {max_size} models, got {squad_size}",
            max_size=max_size, current_size=squad_size,
        )
    return None
```

## Тесты

```python
def test_valid_roster(boyz, warboss, weirdboy):
    units = [("Warboss", 1), ("Boyz", 10), ("Weirdboy", 1)]
    result = validate_roster(units, registry)
    assert result.is_valid
    assert result.total_pts > 0

def test_pts_exceeded():
    too_expensive = [("Warboss", 1)] * 100  # заведомо > 2000pts
    result = validate_roster(too_expensive, registry)
    assert not result.is_valid
    assert any(e.code == "pts_exceeded" for e in result.errors)

def test_no_warlord():
    units = [("Boyz", 10)]  # Boyz can't be warlord
    result = validate_roster(units, registry)
    assert not result.is_valid
    assert any(e.code == "no_warlord" for e in result.errors)

def test_too_many_copies():
    units = [("Warboss", 1)] * 4  # 4 Warbosses, max 3
    result = validate_roster(units, registry)
    assert not result.is_valid
    assert any(e.code == "too_many_copies" for e in result.errors)
```
