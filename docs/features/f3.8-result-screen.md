## JavaScript (Chart.js + Alpine.js)

```javascript
// web/static/result_chart.js

function resultScreen() {
    return {
        result: null,
        gameId: null,
        chart: null,

        get phaseTable() {
            if (!this.result?.round_logs) return [];
            const rows = [];
            for (const log of this.result.round_logs) {
                for (const [phase, events] of Object.entries(log.phase_logs || {})) {
                    let dmg_p1 = 0, dmg_p2 = 0, kills = 0;
                    for (const evt of events) {
                        if (evt.player === 1) dmg_p1 += evt.damage || 0;
                        else dmg_p2 += evt.damage || 0;
                        if (evt.event_type === 'kill') kills++;
                    }
                    rows.push({
                        round: log.round,
                        phase: phase,
                        events: events.length,
                        damage_p1: dmg_p1,
                        damage_p2: dmg_p2,
                        kills: kills,
                    });
                }
            }
            return rows;
        },

        get phaseTotals() {
            const t = {events: 0, damage_p1: 0, damage_p2: 0, kills: 0};
            for (const row of this.phaseTable) {
                t.events += row.events;
                t.damage_p1 += row.damage_p1;
                t.damage_p2 += row.damage_p2;
                t.kills += row.kills;
            }
            return t;
        },

        get destroyedUnits() {
            const result = {1: [], 2: []};
            if (!this.result?.round_logs) return result;
            for (const log of this.result.round_logs) {
                for (const evt of log.events || []) {
                    if (evt.event_type === 'kill' && evt.target_name) {
                        result[evt.player === 1 ? 2 : 1].push(evt.target_name);
                    }
                }
            }
            return result;
        },

        async loadResult(gameId) {
            this.gameId = gameId;
            const resp = await fetch(`/api/results/${gameId}`);
            this.result = await resp.json();
            this.$nextTick(() => this.initChart());
        },

        initChart() {
            const canvas = document.getElementById('vp-chart');
            if (!canvas || !this.result?.round_logs) return;

            const labels = this.result.round_logs.map(l => `R${l.round}`);
            const vp1 = [];
            const vp2 = [];
            let cumulative1 = 0, cumulative2 = 0;

            for (const log of this.result.round_logs) {
                const state = log.end_state?.victory_points || {};
                cumulative1 = state[1] ?? cumulative1;
                cumulative2 = state[2] ?? cumulative2;
                vp1.push(cumulative1);
                vp2.push(cumulative2);
            }

            if (this.chart) this.chart.destroy();

            this.chart = new Chart(canvas, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [
                        {
                            label: 'Player 1',
                            data: vp1,
                            borderColor: '#4ade80',
                            backgroundColor: 'rgba(74, 222, 128, 0.1)',
                            fill: true,
                            tension: 0.3,
                        },
                        {
                            label: 'Player 2',
                            data: vp2,
                            borderColor: '#f87171',
                            backgroundColor: 'rgba(248, 113, 113, 0.1)',
                            fill: true,
                            tension: 0.3,
                        },
                    ],
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { position: 'top' },
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            title: { display: true, text: 'Victory Points' },
                        },
                        x: {
                            title: { display: true, text: 'Round' },
                        },
                    },
                },
            });
        },

        exportJSON() {
            const blob = new Blob(
                [JSON.stringify(this.result, null, 2)],
                {type: 'application/json'}
            );
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `battle-result-${this.gameId}.json`;
            a.click();
            URL.revokeObjectURL(url);
        },
    };
}
```

```python
@router.get("/api/results/{game_id}")
async def get_result(game_id: str,
                     db: sqlite3.Connection = Depends(get_db)):
    """Получить результат симуляции по game_id."""
    replay = load_replay(db, game_id)
    if not replay:
        raise HTTPException(status_code=404, detail="Result not found")
    return asdict(replay)


@router.get("/result/{game_id}")
async def result_page(game_id: str,
                      request: Request,
                      db: sqlite3.Connection = Depends(get_db)):
    """HTML страница результатов."""
    replay = load_replay(db, game_id)
    if not replay:
        raise HTTPException(status_code=404, detail="Result not found")
    return templates.TemplateResponse(
        "result.html",
        {"request": request, "game_id": game_id,
         "result": asdict(replay)},
    )
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Итог ничья | Показать "Draw!", обе армии выделены серым |
| 0 событий в раунде | Phase table показывает пустую строку с 0 |
| Все юниты одной стороны убиты | Destroyed units полный, VP для проигравшего = 0 |
| Результат загружается (pending) | Показать skeleton/spinner |
| Chart.js не загрузился (CDN down) | Показать статическую таблицу VP вместо графика |
| Нет phase_log в replay | Phase table пустая, показать "No data" |
| game_id не существует | 404 страница |
| Огромное количество убитых (>50) | Scrollable список, сгруппировать |
| Экспорт JSON при пустом результате | Скачать пустой JSON `{}` |
