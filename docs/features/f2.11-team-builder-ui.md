# F2.11: Team Builder UI — Faction Picker, Unit Modal, PTS Bar

**Файлы:** `web/templates/team_builder.html`, `web/static/team_builder.js`
**Зависимости:** F2.9 (Roster Validation), F2.10 (Roster CRUD), F1.8 (Icon Map)
**Часы:** 8h

**Цель (SMART):** Создать веб-интерфейс сборщика армии: выбор фракции -> добавление юнитов -> PTS счётчик -> сохранение. Критерий: Alpine.js управляет состоянием (список юнитов, PTS, лимиты), PTS bar обновляется в реальном времени, сохранение через POST /api/rosters, ошибки валидации отображаются под формой. 8 часов.

## Alpine.js Controller

```javascript
// web/static/team_builder.js
function teamBuilder() {
    return {
        // State
        faction: '',
        detachment: '',
        name: 'My Army',
        ptsLimit: 2000,
        roster: [],
        categories: ['HQ', 'Battleline', 'Infantry', 'Vehicle', 'Monster', 'Dedicated Transport'],
        selectedCategory: 'Battleline',
        showModal: false,
        selectedUnit: null,
        squadSize: 1,
        validationErrors: [],

        // Computed
        get unitsByCategory() {
            // From window._unitData grouped by category
            return window._unitData || {};
        },
        get filteredUnits() {
            const cat = this.selectedCategory;
            return (this.unitsByCategory[cat] || [])
                .filter(u => u.faction === this.faction || !this.faction);
        },
        get totalPts() {
            return this.roster.reduce((sum, u) => sum + u.pts * u.squad_size, 0);
        },
        get unitCount() {
            return this.roster.length;
        },
        get isValid() {
            return this.totalPts <= this.ptsLimit
                && this.totalPts > 0
                && this.faction
                && this.roster.length > 0;
        },

        // Methods
        loadUnits() {
            fetch(`/api/units?faction=${this.faction}`)
                .then(r => r.json())
                .then(data => { window._unitData = data; });
        },
        openUnitModal(unit) {
            this.selectedUnit = unit;
            this.squadSize = unit.min_squad || 5;
            this.showModal = true;
        },
        addUnit() {
            if (!this.selectedUnit) return;
            this.roster.push({
                name: this.selectedUnit.name,
                squad_size: this.squadSize,
                pts: this.selectedUnit.points,
            });
            this.showModal = false;
            this.validationErrors = [];
        },
        removeUnit(idx) {
            this.roster.splice(idx, 1);
        },
        saveRoster() {
            fetch('/api/rosters', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    name: this.name,
                    faction: this.faction,
                    detachment: this.detachment,
                    units: this.roster,
                }),
            })
            .then(r => r.json().then(data => ({status: r.status, data})))
            .then(({status, data}) => {
                if (status === 200) {
                    alert('Roster saved!');
                } else if (data.validation_errors) {
                    this.validationErrors = data.validation_errors;
                }
            });
        },
    };
}
```
