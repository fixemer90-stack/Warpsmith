# F1.12: MultiAttack — несколько оружий + отряды
**Цель (SMART):** Расширить симуляцию до SquadAttack: несколько оружий × несколько моделей + dead model removal. Критерий: 9×Shoota + 1×Big Shoota → корректный models_killed, overflow не переносится, wipe_probability рассчитывается. 3 часа.

**Файл:** `backend/engine/combat.py` (расширение)
**Зависимости:** F1.6 (Combat Sequence), F1.7 (Modifiers)

```python
    """Результат симуляции отряд vs отряд."""
    mean_damage: float
    mean_models_killed: float
    mean_wounds_on_last: float
    kill_probability: float               # P(убить хотя бы 1 модель)
    wipe_probability: float               # P(убить весь отряд)

    kills_distribution: list[tuple[int, float]]   # PMF kills
```

## Core Functions

```python
def simulate_squad_volley(
    attacks: list[SquadAttackConfig],
    defender: SquadDefenseConfig,
    modifiers: Optional[list[Modifier]] = None,
    pool: Optional[DicePool] = None,
    n_iterations: int = 10000,
) -> SquadCombatResult:
    """
    Полная симуляция стрельбы отряда.

    Алгоритм:
    1. Для каждого SquadAttack: simulate_weapon_attack() по отдельности
       с n_iterations итераций
    2. Для каждой итерации: суммировать damage по всем оружиям
    3. Распределить damage по моделям (dead model removal)
    4. Агрегировать статистики
    """
    if pool is None:
        pool = DicePool(seed=42)

    # 1. Simulate each weapon type
    per_weapon = []
    for attack_conf in attacks:
        for _ in range(attack_conf.models_firing):
            result = simulate_weapon_attack(
                weapon=attack_conf.weapon,
                attacker=None,  # TODO: pass attacker unit
                defender=defender.unit,
                modifiers=modifiers + defender.modifiers,
                pool=pool,
                n_iterations=n_iterations,
            )
            per_weapon.append(result)

    # 2. Combine damage arrays
    total_damage = sum(r.damage_per_iter for r in per_weapon)

    # 3. Dead model removal per iteration
    kills = np.zeros(n_iterations, dtype=int)
    leftover = np.zeros(n_iterations, dtype=int)
    for i in range(n_iterations):
        k, left = apply_damage_to_squad(
            damage_sequence=[int(r.damage_per_iter[i]) for r in per_weapon],
            squad_size=defender.squad_size,
            wounds_per_model=defender.unit.wounds,
        )
        kills[i] = k
        leftover[i] = left

    # 4. Statistics
    total_stats = compute_stats(total_damage, target_wounds=defender.unit.wounds)
    kill_stats = compute_stats(kills, target_wounds=1)

    return SquadCombatResult(
        mean_damage=total_stats.mean,
        mean_models_killed=float(np.mean(kills)),
        mean_wounds_on_last=float(np.mean(leftover)),
        kill_probability=float(np.mean(kills >= 1)),
        wipe_probability=float(np.mean(kills >= defender.squad_size)),
        damage_distribution=total_stats.pmf,
        kills_distribution=kill_stats.pmf,
        per_weapon_stats=[r.stats for r in per_weapon],
    )


def apply_damage_to_squad(
    damage_sequence: list[int],
    squad_size: int,
    wounds_per_model: int,
) -> tuple[int, int]:
    """
    Распределить урон по моделям отряда.

    Правила 10ed:
    - Каждая модель умирает после wounds_per_model урона
    - Избыточный урон не переносится на следующую модель
    - Урон применяется последовательно: каждый успешный удар → модели

    Returns: (models_killed, wounds_on_last_model)
    """
    if squad_size == 0 or wounds_per_model == 0:
        return (0, 0)

    total_wounds = squad_size * wounds_per_model
    applied = sum(damage_sequence)

    # Не может убить больше моделей чем есть
    max_killable = min(squad_size, applied // wounds_per_model)
    models_killed = max_killable
    wounds_on_last = applied - (models_killed * wounds_per_model)

    return (models_killed, wounds_on_last)


def squad_attack_expected(
    attacks: list[SquadAttackConfig],
    defender: SquadDefenseConfig,
) -> dict:
    """Детерминированные ожидания (для быстрого превью)."""
    total_mean_damage = 0.0
    for ac in attacks:
        w = ac.weapon
        hit_p = (7 - w.skill) / 6
        wound_target = _toughness_table(w.strength, defender.unit.toughness)
        wound_p = (7 - wound_target) / 6
        save_target = defender.unit.best_save(w.ap)
        save_p = (7 - save_target) / 6

        # Средний дамаг на атаку
        num_dice, sides, bonus = w.damage_dice
        avg_dmg = bonus if num_dice == 0 else (num_dice * (sides + 1) / 2 + bonus)

        per_shot = hit_p * wound_p * (1 - save_p) * avg_dmg
        total_mean_damage += per_shot * ac.models_firing

    expected_kills = total_mean_damage / defender.unit.wounds
    return {
        "expected_damage": round(total_mean_damage, 2),
        "expected_kills": round(expected_kills, 2),
    }
```

## Тесты

```python
def test_apply_damage_basic():
    """5 Marines (W2), total 10 wounds. Deal 6 damage."""
    killed, left = apply_damage_to_squad([1, 3, 2], squad_size=5, wounds_per_model=2)
    assert killed == 3  # 1+3+2 = 6 → 3 × 2 = 6 → 3 dead
    assert left == 0

def test_apply_damage_overflow():
    """Урон не переносится."""
    killed, left = apply_damage_to_squad([3, 1], squad_size=2, wounds_per_model=2)
    # 3 dmg → 1 dead (2 wounds), 1 overflow. 1 dmg → 0 dead
    assert killed == 1
    assert left == 1  # 1 wound remaining

def test_no_damage():
    killed, left = apply_damage_to_squad([], squad_size=5, wounds_per_model=2)
    assert killed == 0
    assert left == 0

def test_squad_simulate_10boyz_vs_5marines(seeded_pool):
    shoota = Weapon(name="Shoota", ..., attacks_dice=(0,0,3), skill=5, ...)
    marine = Unit(name="Tactical Marine", toughness=4, save=3, wounds=2, ...)

    attacks = [SquadAttackConfig(shoota, 9)]
    defender = SquadDefenseConfig(marine, squad_size=5)

    result = simulate_squad_volley(attacks, defender, pool=seeded_pool, n_iterations=10000)
    assert result.mean_models_killed >= 0
    assert result.mean_damage > 0
    # 9 × 3 × 0.33 × 0.5 × 0.33 × 1 = ~1.47 mean damage → ~0.73 kills
    assert result.wipe_probability < 0.1  # очень маловероятно убить 5 Marines
```
