# F3.5: Auto-play — AI vs AI Full Scenario

**Файл:** `backend/engine/ai/autoplay.py` (создать)
**Зависимости:** F2.5 (Game Loop), F3.1 (Decision Engine), F3.2 (Ork AI), F3.3 (T'au AI), F3.4 (Deployment AI), F2.4 (Missions)
**Часы:** 6h

**Цель (SMART):** Реализовать AutoPlay, запускающий полную симуляцию AI vs AI: deploy units → run 5 rounds → return replay log. Критерий: вызов run_auto_game(roster_a, roster_b, mission) проходит без ошибок, возвращает AutoPlayResult с полным логом (все фазы всех раундов), не падает при любых комбинациях фракций. 6 часов.

## Data Model

```python
from dataclasses import dataclass, field
from typing import Optional
from backend.engine.ai.deployment import DeploymentType, place_units
from backend.engine.ai.ork import OrkAI
from backend.engine.ai.tau import TauAI
from backend.state.game_state import GameState, RosterState
from backend.engine.scenario import run_round, RoundLog


class AutoPlayError(Exception):
    """Базовое исключение AutoPlay."""


class InvalidRosterError(AutoPlayError):
    """Ростер не прошёл валидацию."""


class TimeoutError(AutoPlayError):
    """Симуляция превысила лимит времени."""


@dataclass
class AutoPlayConfig:
    """Конфигурация симуляции."""
    max_rounds: int = 5
    deployment_type: DeploymentType = DeploymentType.STANDARD
    seed: int = 42
    time_limit_seconds: int = 30
    use_sticky_objectives: bool = True
    enable_stratagems: bool = True
    log_all_events: bool = True


@dataclass
class AutoPlayResult:
    """Результат полной симуляции."""
    game_state: GameState
    round_logs: list[RoundLog]
    placements: dict[int, list]  # player → [Placement]
    error: Optional[str] = None
    total_duration_ms: float = 0.0
    summary: dict = field(default_factory=dict)

    def to_dict(self) -> dict:
        """Сериализация результата для JSON."""
        return {
            "rounds": len(self.round_logs),
            "victory_points": dict(self.game_state.victory_points),
            "winner": self._determine_winner(),
            "summary": self.summary,
            "error": self.error,
            "duration_ms": self.total_duration_ms,
        }

    def _determine_winner(self) -> Optional[int]:
        vp = self.game_state.victory_points
        if vp[1] > vp[2]:
            return 1
        elif vp[2] > vp[1]:
            return 2
        return None  # ничья
```

## AI Resolver

```python
def resolve_ai_for_faction(faction: str) -> object:
    """Вернуть AI класс для фракции.

    Registry расширяемый: при добавлении новой фракции
    нужно зарегистрировать AI класс здесь.

    Orks → OrkAI
    T'au → TauAI
    default → DecisionEngine (generic greedy)
    """
    registry = {
        "orks": OrkAI,
        "tau": TauAI,
        "tyranids": None,  # TODO: F3.9
        "space_marines": None,  # TODO: F4.x
    }
    cls = registry.get(faction.lower())
    if cls is None:
        from backend.engine.ai.decision import DecisionEngine
        return DecisionEngine()
    return cls()
```

## Main AutoPlay Loop

```python
def run_auto_game(roster_a: RosterState,
                  roster_b: RosterState,
                  mission: "Mission",
                  config: AutoPlayConfig = None) -> AutoPlayResult:
    """Запустить полную AI vs AI симуляцию.

    Flow:
    1. Валидация ростеров
    2. Создание GameState
    3. Deploy юнитов на карту
    4. Создание AI инстансов
    5. Запуск game loop (run_round × max_rounds)
    6. Сбор логов и summary
    7. Возврат результата
    """
    import time
    start = time.time()
    config = config or AutoPlayConfig()

    # 1. Validate
    errors = _validate_rosters(roster_a, roster_b)
    if errors:
        return AutoPlayResult(
            game_state=None, round_logs=[],
            placements={},
            error=f"Roster validation failed: {errors}",
        )

    # 2. Create game state
    map_ = create_default_map(seed=config.seed)
    state = GameState(
        roster_a=roster_a,
        roster_b=roster_b,
        seed=config.seed,
        mission=mission.name,
    )

    # 3. Deploy
    deployments = {}
    for player, roster in [(1, roster_a), (2, roster_b)]:
        placements = place_units(roster, config.deployment_type,
                                 map_, mission.objectives)
        deployments[player] = placements
        for p in placements:
            unit = _find_unit_in_roster(roster, p.unit_id)
            if unit:
                unit.position.x = p.x
                unit.position.y = p.y

    # 4. Create AIs
    ai_a = resolve_ai_for_faction(roster_a.faction)
    ai_b = resolve_ai_for_faction(roster_b.faction)

    # 5. Game loop
    round_logs = []
    try:
        for r in range(config.max_rounds):
            # Time check
            elapsed = (time.time() - start) * 1000
            if elapsed > config.time_limit_seconds * 1000:
                raise TimeoutError(
                    f"Simulation exceeded {config.time_limit_seconds}s"
                )

            log = RoundLog(round=r + 1)
            state = run_round(state, map_, mission, log,
                              ai_a=ai_a, ai_b=ai_b,
                              config=config)
            round_logs.append(log)

            # End condition check
            if _check_game_end(state, mission):
                break

    except AutoPlayError as e:
        return AutoPlayResult(
            game_state=state, round_logs=round_logs,
            placements=deployments, error=str(e),
            total_duration_ms=(time.time() - start) * 1000,
        )

    # 6. Summary
    summary = _build_summary(state, round_logs, deployments)

    return AutoPlayResult(
        game_state=state, round_logs=round_logs,
        placements=deployments,
        total_duration_ms=(time.time() - start) * 1000,
        summary=summary,
    )


def _build_summary(state: GameState, round_logs: list[RoundLog],
                   deployments: dict) -> dict:
    """Построить сводку симуляции."""
    total_kills = {1: 0, 2: 0}
    total_damage = {1: 0, 2: 0}
    charge_count = {1: 0, 2: 0}

    for log in round_logs:
        for event in log.events:
            player = event.get("player", 0)
            if event.get("type") == "kill":
                total_kills[player] = total_kills.get(player, 0) + 1
            if event.get("damage", 0) > 0:
                total_damage[player] = total_damage.get(player, 0) + event["damage"]
            if event.get("action") == "charge":
                charge_count[player] = charge_count.get(player, 0) + 1

    return {
        "victory_points": dict(state.victory_points),
        "winner": _determine_winner(state),
        "total_kills": total_kills,
        "total_damage": total_damage,
        "charge_count": charge_count,
        "rounds_played": len(round_logs),
        "army_a": {
            "faction": state.roster_a.faction,
            "pts": state.roster_a.total_pts,
        },
        "army_b": {
            "faction": state.roster_b.faction,
            "pts": state.roster_b.total_pts,
        },
    }
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Ростер невалиден (превышение PTS) | Вернуть InvalidRosterError, не запускать симуляцию |
| Оба AI одинаковые (Ork vs Ork) | Работает корректно, каждый AI независим |
| Юнит без оружия | Пропустить shooting phase |
| Все юниты одной стороны уничтожены | Завершить игру досрочно |
| Timeout превышен | Вернуть частичный результат с error |
| Миссия с 0 objectives | deployment без objective scoring |
| Seed = 0 (random) | Работает, но невоспроизводимо |
| Фракция без AI (например, Space Marines) | Использовать generic DecisionEngine |
| Ростер с 1 юнитом | Не падать, игра завершается за 1 раунд |

```python
def test_run_auto_game_basic():
    """Полная симуляция Orks vs Tau завершается без ошибок."""
    roster_a = make_ork_roster(pts=1000)
    roster_b = make_tau_roster(pts=1000)
    mission = make_mission("only_war")
    result = run_auto_game(roster_a, roster_b, mission)
    assert result.error is None
    assert len(result.round_logs) >= 1


def test_auto_game_returns_result():
    result = run_auto_game(make_ork_roster(), make_tau_roster(),
                           make_mission())
    assert isinstance(result, AutoPlayResult)
    assert result.summary["winner"] in [1, 2, None]


def test_invalid_roster_returns_error():
    roster_a = make_roster(pts=9999)  # превышение
    result = run_auto_game(roster_a, make_tau_roster(), make_mission())
    assert result.error is not None
    assert "validation" in result.error.lower()


def test_single_unit_roster():
    roster = make_roster(units=1, pts=500)
    result = run_auto_game(roster, make_tau_roster(), make_mission())
    assert result.error is None


def test_timeout_returns_partial():
    config = AutoPlayConfig(time_limit_seconds=0.001)
    result = run_auto_game(make_ork_roster(), make_tau_roster(),
                           make_mission(), config)
    assert result.error is not None
    assert "timeout" in result.error.lower()
    assert len(result.round_logs) < 5  # не все раунды


def test_reproducible_with_seed():
    config = AutoPlayConfig(seed=42)
    result1 = run_auto_game(make_ork_roster(), make_tau_roster(),
                            make_mission(), config)
    result2 = run_auto_game(make_ork_roster(), make_tau_roster(),
                            make_mission(), config)
    assert result1.summary["winner"] == result2.summary["winner"]


def test_ork_vs_ork():
    result = run_auto_game(make_ork_roster(), make_ork_roster(),
                           make_mission())
    assert result.error is None


def test_unknown_faction_uses_default_ai():
    roster = make_roster(faction="unknown_faction")
    result = run_auto_game(roster, make_tau_roster(), make_mission())
    assert result.error is None


def test_auto_game_duration_measured():
    result = run_auto_game(make_ork_roster(), make_tau_roster(),
                           make_mission())
    assert result.total_duration_ms > 0
```
