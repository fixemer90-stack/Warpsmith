# F1.10: PMF Chart — график распределения урона
**Цель (SMART):** Визуализировать PMF через Chart.js bar chart с цветовой индикацией (kill threshold = красный). Критерий: HTMX-подгрузка partial-шаблона после /simulate, 4 stat-блока (mean, P(kill), std, time), canvas#pmf-chart, tooltip с процентами. 4 часа.

**Файлы:**
- `web/templates/partials/sim_chart.html` (создать)
- `web/static/sim_chart.js` (создать)

**Зависимости:** F1.9 (API endpoint), Chart.js (CDN)
**Часы:** 4h

## Chart.js config

```javascript
// web/static/sim_chart.js

class DamageChart {
    constructor(canvasId) {
        this.ctx = document.getElementById(canvasId)?.getContext('2d');
        this.chart = null;
        this.cdfChart = null;
    }

    render(data) {
        if (!this.ctx) return;

        const labels = data.results.pmf_by_damage.map(
            entry => `${entry.damage} dmg`
        );
        const values = data.results.pmf_by_damage.map(
            entry => entry.probability
        );
        const killThreshold = data.results.mean_damage > 0
            ? this._findKillThreshold(data) : null;

        // Destroy previous chart
        if (this.chart) this.chart.destroy();

        this.chart = new Chart(this.ctx, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Probability',
                    data: values,
                    backgroundColor: values.map((v, i) => {
                        const dmg = data.results.pmf_by_damage[i].damage;
                        if (dmg === 0) return 'rgba(156, 163, 175, 0.7)'; // gray
                        if (killThreshold !== null && dmg >= killThreshold) {
                            return 'rgba(239, 68, 68, 0.8)'; // red (kill)
                        }
                        return 'rgba(59, 130, 246, 0.7)'; // blue (damage)
                    }),
                    borderColor: values.map((v, i) => {
                        const dmg = data.results.pmf_by_damage[i].damage;
                        if (dmg === 0) return 'rgba(156, 163, 175, 1)';
                        if (killThreshold !== null && dmg >= killThreshold) {
                            return 'rgba(239, 68, 68, 1)';
                        }
                        return 'rgba(59, 130, 246, 1)';
                    }),
                    borderWidth: 1,
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    tooltip: {
                        callbacks: {
                            label: (ctx) => {
                                const pct = (ctx.parsed.y * 100).toFixed(1);
                                const dmg = data.results.pmf_by_damage[ctx.dataIndex].damage;
                                let label = `${dmg} damage: ${pct}%`;
                                if (killThreshold !== null && dmg >= killThreshold) {
                                    label += ' ☠️ kill';
                                }
                                return label;
                            }
                        }
                    },
                    annotation: killThreshold ? {
                        annotations: {
                            killLine: {
                                type: 'line',
                                xMin: killThreshold - 0.5,
                                xMax: killThreshold - 0.5,
                                borderColor: 'rgba(239, 68, 68, 0.5)',
                                borderWidth: 2,
                                borderDash: [5, 5],
                                label: {
                                    display: true,
                                    content: `Kill (${killThreshold}+ dmg)`,
                                    position: 'start',
                                }
                            }
                        }
                    } : {},
                },
                scales: {
                    y: {
                        beginAtZero: true,
                        max: 1.0,
                        title: { display: true, text: 'Probability' },
                        ticks: {
                            callback: (v) => `${(v * 100).toFixed(0)}%`
                        }
                    },
                    x: {
                        title: { display: true, text: 'Damage' }
                    }
                }
            }
        });
    }

    _findKillThreshold(data) {
        // Ищем damage = wounds of target (убийство)
        // Из meta или расчёт по damage_per_success
        return 2; // default: marine wounds
    }
}

// Init on HTMX swap
document.addEventListener('htmx:afterSwap', (event) => {
    if (event.detail.target.id === 'chart-container') {
        const chart = new DamageChart('pmf-chart');
        // data хранится в data-атрибуте или window
        if (window._lastSimData) {
            chart.render(window._lastSimData);
        }
    }
});
```
