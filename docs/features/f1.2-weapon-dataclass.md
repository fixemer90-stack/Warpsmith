## Dice Expression Parser

```python
import re

def parse_dice_expression(expr: str) -> DiceExpr:
    """
    Парсит строку атак/дамага в (num_dice, sides, bonus).

    >>> parse_dice_expression("D6")     → (1, 6, 0)
    >>> parse_dice_expression("2D3")    → (2, 3, 0)
    >>> parse_dice_expression("D6+2")   → (1, 6, 2)
    >>> parse_dice_expression("3")      → (0, 0, 3)  # fixed
    >>> parse_dice_expression("2D3+1")  → (2, 3, 1)
    >>> parse_dice_expression("Melee")  → (0, 0, 0)  # special
    """
    m = re.match(r'(?:(\d+))?[Dd](\d+)(?:\s*\+\s*(\d+))?', expr.strip())
    if m:
        num = int(m.group(1)) if m.group(1) else 1
        sides = int(m.group(2))
        bonus = int(m.group(3)) if m.group(3) else 0
        return (num, sides, bonus)
    # fixed number
    m2 = re.match(r'(\d+)$', expr.strip())
    if m2:
        return (0, 0, int(m2.group(1)))
    return (0, 0, 0)
```

## Resolution helpers

```python
def resolve_dice(dice: DiceExpr, rng: np.random.Generator) -> int:
    """Бросок кубов по DiceExpr."""
    num, sides, bonus = dice
    if num == 0:
        return bonus  # fixed
    total = int(rng.integers(1, sides + 1, size=num).sum())
    return total + bonus

def dice_expr_to_str(dice: DiceExpr) -> str:
    num, sides, bonus = dice
    if num == 0:
        return str(bonus)
    base = f"{num if num > 1 else ''}D{sides}"
    return f"{base}+{bonus}" if bonus else base
```

## Тесты

```python
def test_parse_dice():
    assert parse_dice_expression("D6") == (1, 6, 0)
    assert parse_dice_expression("2D3") == (2, 3, 0)
    assert parse_dice_expression("D6+2") == (1, 6, 2)
    assert parse_dice_expression("3") == (0, 0, 3)
    assert parse_dice_expression("2D3+1") == (2, 3, 1)

def test_resolve_dice():
    rng = np.random.default_rng(42)
    d = resolve_dice((1, 6, 0), rng)
    assert 1 <= d <= 6
    d = resolve_dice((0, 0, 3), rng)
    assert d == 3
```
