# web/routes/pages.py

@router.get("/simulate", name="simulate_page")
async def simulate_page(request: Request):
    """Страница симулятора: форма выбора оружия и цели."""
    registry.load()
    units = registry.list_units()
    # Группировка по фракциям для select
    units_by_faction = defaultdict(list)
    for name in units:
        unit = registry.get_unit(name)
        units_by_faction[unit.faction].append(unit)

    return templates.TemplateResponse(
        "round_viewer.html",
        {
            "request": request,
            "units_by_faction": dict(units_by_faction),
            "available_modifiers": [
                "heavy", "lethal_hits", "sustained_hits_1",
                "devastating_wounds", "ignores_cover",
            ],
        },
    )


@router.post("/simulate", name="simulate_result")
async def simulate_result(
    request: Request,
    weapon_id: str = Form(...),
    attacker_id: str = Form(...),
    defender_id: str = Form(...),
    n_simulations: int = Form(default=10000),
    modifiers: list[str] = Form(default=[]),
    range_bracket: str = Form(default="long"),
):
    """HTMX partial: результат симуляции."""
    # Вызов API (переиспользовать логику из api.py)
    try:
        result = await run_simulation(
            weapon_id, attacker_id, defender_id,
            n_simulations, modifiers, range_bracket,
        )
    except ValueError as e:
        return templates.TemplateResponse(
            "partials/sim_error.html",
            {"request": request, "error": str(e)},
        )

    return templates.TemplateResponse(
        "partials/sim_chart.html",
        {
            "request": request,
            "results": result["results"],
            "steps": result["results"]["steps"],
            "simulation_time_ms": result["results"]["simulation_time_ms"],
            "raw_json": json.dumps(result, indent=2),
            "full_response": result,
        },
    )
```

<script>
function simulatorForm() {
    return {
        attackerId: '',
        weaponId: '',
        weapons: [],
        updateWeapons() {
            // HTMX + Alpine: получить список оружия для attacker
            if (!this.attackerId) {
                this.weapons = [];
                return;
            }
            fetch(`/api/units/${this.attackerId}/weapons`)
                .then(r => r.json())
                .then(data => { this.weapons = data.weapons; });
        }
    };
}

# web/routes/api.py — доп. эндпоинт для UI

@router.get("/api/units/{unit_id}/weapons")
async def list_unit_weapons(unit_id: str):
    """Список оружий юнита для UI."""
    unit = registry.get_unit(unit_id)
    if not unit:
        raise HTTPException(404)
    weapons = []
    for w in unit.ranged_weapons:
        weapons.append({"name": w.name, "type": "ranged", "profile": ...})
    for w in unit.melee_weapons:
        weapons.append({"name": w.name, "type": "melee", ...})
    return {"weapons": [w["name"] for w in weapons]}
```
