## Guided Fire System

```python
@dataclass
class GuidedFirePair:
    """Пара Spotter → Firer для Guided Fire."""
    spotter_id: str
    firer_id: str
    target_id: str
    ignore_cover: bool = True
    bs_bonus: int = 1  # +1 to hit


def build_guided_fire_pairs(roster: RosterState,
                            opponent: RosterState,
                            map_: GameMap) -> list[GuidedFirePair]:
    """Построить пары Spotter-Firer.

    Spotter должен:
    - Иметь keyword 'MARKERLIGHT' или способность 'Spotter'
    - Видеть цель (LoS)
    - Быть в range своей споттер-способности (обычно 18-24")

    Firer должен:
    - Видеть ту же цель
    - Иметь keyword 'FOR_THE_GREATER_GOOD'
    """
    pairs = []
    spotters = [u for u in roster.units
                if 'MARKERLIGHT' in u.keywords or 'spotter' in u.abilities]
    firers = [u for u in roster.units
              if 'FOR_THE_GREATER_GOOD' in u.keywords]

    for spotter in spotters:
        for target in opponent.units:
            if target.squad_size <= 0:
                continue
            if not has_los(map_, spotter.position, target.position):
                continue
            # Найти лучшего firer для этой цели
            best_firer = _find_best_firer(spotter, target, firers, map_)
            if best_firer:
                pairs.append(GuidedFirePair(
                    spotter_id=spotter.unit_id,
                    firer_id=best_firer.unit_id,
                    target_id=target.unit_id,
                ))
    return pairs


def _find_best_firer(spotter: UnitState, target: UnitState,
                     firers: list[UnitState],
                     map_: GameMap) -> Optional[UnitState]:
    """Найти лучшего firer: в range, LoS, максимальный ожидаемый урон."""
    best = None
    best_score = -1
    for firer in firers:
        if firer.squad_size <= 0:
            continue
        if not has_los(map_, firer.position, target.position):
            continue
        if not _in_range_of_weapons(firer, target):
            continue
        # Оценка по урону
        dmg = estimate_guided_damage(firer, target)
        if dmg > best_score:
            best_score = dmg
            best = firer
    return best
```

## Kauyon / Mont'ka Management

```python
class TauDoctrine(Enum):
    NONE = "none"
    MONTKA = "montka"    # раунды 1-3: агрессивное наступление
    KAUYON = "kauyon"    # раунды 3-5: засада, перезарядка


def choose_doctrine(state: GameState, round_num: int) -> TauDoctrine:
    """Выбор доктрины T'au.

    Mont'ka (Ранняя игра):
    - Раунды 1-3
    - +1 к Wound при стрельбе в ближней дистанции (< 12")
    - Возможность Advance + Shoot без штрафа

    Kauyon (Поздняя игра):
    - Раунды 3-5
    - Переброс 1-й попаданий (Sustained Hits на выбор)
    - Требует выжидания
    """
    if round_num <= 2:
        # Mont'ka: агрессивный старт
        if _has_markerlight_coverage(state, threshold=0.6):
            return TauDoctrine.MONTKA
    elif round_num >= 3:
        # Kauyon: засада
        if _has_units_in_cover(state, threshold=0.7):
            return TauDoctrine.KAUYON
    return TauDoctrine.NONE


def apply_doctrine_bonus(damage_result: dict, doctrine: TauDoctrine,
                         distance: float) -> dict:
    """Применить бонусы доктрины к результатам стрельбы."""
    if doctrine == TauDoctrine.MONTKA and distance <= 12:
        # +1 к wound roll (Half-range)
        damage_result["wound_bonus"] = 1
    elif doctrine == TauDoctrine.KAUYON:
        # Sustained Hits 1 на all hits
        damage_result["sustained_hits"] = 1
    return damage_result
```

## Tau-specific Scoring Overrides

```python
def tau_score_shoot(actor: UnitState, target: UnitState,
                    weapon_idx: int, ctx: EvaluationContext,
                    guided_pairs: list[GuidedFirePair]) -> float:
    """T'au shooting с Guided Fire.

    - Guided: +1 to hit, ignore cover
    - Markerlight: +1 to hit
    - Drone support: extra wounds
    """
    base = score_shoot(actor, target, weapon_idx, ctx)

    # Guided Fire бонус
    if _is_guided(actor, target, guided_pairs):
        base *= 1.4  # +1 to hit + ignore cover

    # Markerlight Drones
    if 'MARKERLIGHT' in actor.keywords:
        base *= 1.2

    # Доктрина бонус
    if ctx.tau_doctrine == TauDoctrine.MONTKA:
        dist = _distance(actor.position, target.position)
        if dist <= 12:
            base *= 1.3  # +1 to wound
    elif ctx.tau_doctrine == TauDoctrine.KAUYON:
        base *= 1.5  # sustained hits — больше урона

    return base


def tau_score_move(actor: UnitState, target_pos: tuple[int, int],
                   ctx: EvaluationContext) -> float:
    """T'au movement — держать дистанцию.

    - Бонус за движение к cover
    - Бонус за увеличение дистанции от врага
    - Штраф за движение в открытое пространство
    """
    base = score_move(actor, target_pos, ctx)

    for enemy in ctx.opponent.units:
        current_dist = _distance(actor.position, enemy.position)
        new_dist = _distance(target_pos, enemy.position)

        # Бонус за увеличение дистанции
        if new_dist > current_dist:
            base += 0.3 * (new_dist - current_dist) / 6.0

        # Штраф если становимся ближе чем 12"
        if new_dist < 12:
            base -= 0.5

    # Cover бонус
    if _has_cover(ctx.map_, target_pos):
        base *= 1.3

    return base
```

## TauAI Class

```python
class TauAI:
    """T'au AI — дистанционный контроль, Guided Fire, доктрины."""

    def __init__(self):
        self.weights = TAU_WEIGHTS
        self.doctrine: TauDoctrine = TauDoctrine.NONE
        self.guided_pairs: list[GuidedFirePair] = []

    def decide(self, unit: UnitState, ctx: EvaluationContext) -> Action:
        """Принять решение за T'au юнит."""
        # Обновить доктрину
        self.doctrine = choose_doctrine(ctx.state, ctx.state.round)

        # Построить Guided Fire пары (кэшировать)
        if not self.guided_pairs:
            self.guided_pairs = build_guided_fire_pairs(
                ctx.state.active_roster(),
                ctx.state.opponent_roster(),
                ctx.map_,
            )

        # Выбрать действие с T'au весами
        action = choose_action(unit, ctx, weights=self.weights)

        # Пост-обработка: если стрельба — проверить Guided Fire
        if action.type == ActionType.SHOOT:
            guided = _is_guided(unit, action.target_id, self.guided_pairs)
            if guided and action.rationale:
                action.rationale += " [Guided Fire]"

        # Пост-обработка: избегать заряда
        if action.type == ActionType.CHARGE:
            # T'au заряжают только если цель ослаблена (< 25% ран)
            target = _find_unit(ctx.opponent, action.target_id)
            if target and target.wounds_remaining > target.max_wounds * 0.25:
                # Отмена заряда — отступить
                retreat_pos = _find_retreat_position(unit, ctx)
                return Action(
                    type=ActionType.FALL_BACK,
                    target_position=retreat_pos,
                    score=action.score * 0.8,
                    rationale="T'au do not charge!",
                )

        return action
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Нет Markerlight юнитов | Guided Fire недоступен, стрельба без бонуса |
| Spotter уничтожен | Guided-пары перестраиваются |
| Kauyon активен, враг в 6" | Предпочесть Fall Back вместо стрельбы |
| Mont'ka активен, враг далеко | Score снижен (no half-range bonus) |
| Все юниты в cover | Выбрать Kauyon |
| Юнит с Drone (Shield Drone) | Учесть extra wound в survival score |
| Враг заряжает T'au | Использовать Overwatch, затем Fall Back |
| Нет valid retreat пути | Стоять и стрелять (no Fall Back available) |
| Guided пара направлена на мёртвую цель | Перестроить, выбрать новую цель |
