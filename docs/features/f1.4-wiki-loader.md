## Parser module (`backend/loader/parser.py`)

```python
"""Парсинг YAML frontmatter + markdown таблиц → Unit/Weapon."""

import re
import frontmatter
from pathlib import Path
from typing import Optional, Any

from backend.model.unit import Unit, Weapon, WargearSlot
from backend.model.unit import parse_dice_expression


# Поля, которые маппятся напрямую из YAML в Unit
DIRECT_FIELDS = {
    "title": "name",
    "faction": "faction",
    "movement": "movement",
    "toughness": "toughness",
    "save": "save",
    "wounds": "wounds",
    "leadership": "leadership",
    "objective_control": "objective_control",
    "points": "points",
    "edition": "edition",
}

# Поля-списки с переименованием
LIST_FIELDS = {
    "keywords": "keywords",
    "faction_keywords": "faction_keywords",
}

# Поля с кастомным парсингом
SPECIAL_FIELDS = {
    "invulnerable_save": lambda v: int(v) if v and v != "none" else None,
    "feel_no_pain": lambda v: int(v) if v and v != "none" else None,
    "model_count": lambda v: tuple(int(x) for x in v.replace(" ", "").split("-"))
                              if isinstance(v, str) else (v[0], v[1]) if isinstance(v, list) else (1, 1),
    "is_epic_hero": bool,
    "can_be_warlord": bool,
    "is_leader": bool,
}


def parse_unit(filepath: Path) -> Optional[Unit]:
    """Парсинг .md файла → Unit."""
    try:
        post = frontmatter.load(str(filepath))
    except Exception as e:
        logger.error(f"Failed to parse {filepath}: {e}")
        return None

    fm = post.metadata
    if not fm.get("title"):
        logger.warning(f"No title in {filepath}")
        return None

    # Faction из пути
    faction = filepath.parent.name

    # Прямые поля
    kwargs: dict[str, Any] = {}
    for yaml_key, obj_attr in DIRECT_FIELDS.items():
        if yaml_key == "title":
            kwargs[obj_attr] = str(fm.get("title", ""))
        elif yaml_key in fm:
            kwargs[obj_attr] = fm[yaml_key]

    # Специальные поля
    for yaml_key, parser in SPECIAL_FIELDS.items():
        if yaml_key in fm:
            kwargs[yaml_key] = parser(fm[yaml_key])

    # Списки
    for yaml_key, obj_attr in LIST_FIELDS.items():
        kwargs[obj_attr] = fm.get(yaml_key, [])

    # Оружие — из frontmatter или markdown таблицы
    weapons_data = fm.get("weapons", [])
    ranged_w, melee_w = _parse_weapons_from_frontmatter(weapons_data)
    kwargs["ranged_weapons"] = ranged_w
    kwargs["melee_weapons"] = melee_w

    # Если weapons нет в frontmatter — парсить markdown таблицу
    if not weapons_data:
        ranged_w, melee_w = _parse_weapons_from_markdown(post.content)
        kwargs["ranged_weapons"] = ranged_w
        kwargs["melee_weapons"] = melee_w

    kwargs["faction"] = faction

    return Unit(**kwargs)


def _parse_weapons_from_frontmatter(data: list[dict]) -> tuple[list[Weapon], list[Weapon]]:
    """Парсинг weapons из YAML frontmatter."""
    ranged = []
    melee = []
    for w in data:
        is_ranged = str(w.get("range", "0")).lower() != "melee"
        weapon = Weapon(
            name=w.get("name", ""),
            type="ranged" if is_ranged else "melee",
            range_max=int(str(w.get("range", "0")).replace('\"', ""))
                     if is_ranged else None,
            attacks_dice=parse_dice_expression(str(w.get("attacks", "1"))),
            skill=int(str(w.get("skill", "5+")).replace("+", "")),
            strength=int(w.get("strength", 3)),
            ap=int(w.get("ap", 0)),
            damage_dice=parse_dice_expression(str(w.get("damage", "1"))),
            tags=w.get("tags", []),
            abilities=w.get("abilities", []),
        )
        if is_ranged:
            ranged.append(weapon)
        else:
            melee.append(weapon)
    return ranged, melee


def _parse_weapons_from_markdown(body: str) -> tuple[list[Weapon], list[Weapon]]:
    """Парсинг weapons из markdown-таблицы.

    Ищет таблицу вида:
    | Weapon | Range | A | BS | S | AP | D | Abilities |
    | Shoota | 18\" | 3 | 5+ | 4 | 0 | 1 | - |
    """
    # Ищем первую таблицу после заголовка Ranged Weapons / Weapons
    table_pattern = r'\|(.+)\|\n\|[-| ]+\|\n((?:\|.+\|\n?)+)'
    match = re.search(table_pattern, body)
    if not match:
        return [], []

    header = match.group(1)
    rows_text = match.group(2)

    # Определяем колонки
    cols = [c.strip().lower() for c in header.split("|") if c.strip()]

    ranged, melee = [], []
    for row in rows_text.strip().split("\n"):
        cells = [c.strip() for c in row.split("|") if c.strip()]
        if len(cells) < 7:
            continue

        name = cells[0]
        range_str = cells[1].replace('\"', "")
        is_ranged = range_str.lower() != "melee"

        weapon = Weapon(
            name=name,
            type="ranged" if is_ranged else "melee",
            range_max=int(range_str) if is_ranged else None,
            attacks_dice=parse_dice_expression(cells[2]),
            skill=int(cells[3].replace("+", "")),
            strength=int(cells[4]),
            ap=int(cells[5]),
            damage_dice=parse_dice_expression(cells[6]),
            abilities=[a.strip() for a in cells[7].split(",")] if len(cells) > 7 else [],
        )
        if is_ranged:
            ranged.append(weapon)
        else:
            melee.append(weapon)

    return ranged, melee
```

## Расширение Registry

```python
class WikiRegistry:
    def __init__(self, wiki_path: str = ""):
        # ... существующий код ...
        self.cache_path = Path.home() / ".cache" / "wiki_registry.pkl"
        self._file_mtimes: dict[str, float] = {}

    def load(self, use_cache: bool = True) -> dict[str, Unit]:
        """Загрузить с кэшированием."""
        if self._loaded:
            return self.units

        if use_cache and self._load_from_cache():
            return self.units

        self.units = self._scan_and_parse()
        self._loaded = True

        if use_cache:
            self._save_cache()

        return self.units

    def _scan_and_parse(self) -> dict[str, Unit]:
        """Полное сканирование всех .md файлов."""
        units: dict[str, Unit] = {}
        units_dir = self.wiki_path / "units"
        if not units_dir.exists():
            return units

        for faction_dir in sorted(units_dir.iterdir()):
            if not faction_dir.is_dir():
                continue
            for file in faction_dir.glob("*.md"):
                unit = parse_unit(file)
                if unit:
                    units[unit.name] = unit
                    self._file_mtimes[str(file)] = file.stat().st_mtime

        logger.info(f"Wiki loaded: {len(units)} units from {len(units)} files")
        return units

    def _load_from_cache(self) -> bool:
        """Загрузить из кэша, если mtime не изменились."""
        try:
            with open(self.cache_path, "rb") as f:
                data = pickle.load(f)
            # Проверить mtime
            for path, mtime in data["mtimes"].items():
                p = Path(path)
                if not p.exists() or p.stat().st_mtime != mtime:
                    return False
            self.units = data["units"]
            self._loaded = True
            logger.info(f"Loaded {len(self.units)} units from cache")
            return True
        except (FileNotFoundError, pickle.UnpicklingError, KeyError):
            return False

    def _save_cache(self) -> None:
        self.cache_path.parent.mkdir(parents=True, exist_ok=True)
        with open(self.cache_path, "wb") as f:
            pickle.dump({"units": self.units, "mtimes": self._file_mtimes}, f)
```

## Обработка ошибок

| Ситуация | Поведение |
|----------|-----------|
| Нет `title` в frontmatter | log warning, skip file |
| Невалидный YAML | log error, skip file |
| Отсутствует `wiki/units/` | log warning, empty registry |
| Непарсибельное значение поля | log warning, default value |
| Пустой файл | skip |

SAMPLE_BOY = """---
title: Boyz
faction: orks
movement: 5
toughness: 5
save: 5
wounds: 2
leadership: 7
objective_control: 2
model_count: 10-20
points: 85
keywords: [Infantry, Battleline, Core, Orks]
faction_keywords: [Orks, Waaagh!]
weapons:
  - name: Shoota
    range: 18
    attacks: "3"
    skill: "5+"
    strength: 4
    ap: 0
    damage: "1"
    tags: [assault]
---
"""

def test_parse_unit_from_frontmatter(tmp_path):
    f = tmp_path / "units" / "orks" / "Boyz.md"
    f.parent.mkdir(parents=True)
    f.write_text(SAMPLE_BOY)
    unit = parse_unit(f)
    assert unit is not None
    assert unit.name == "Boyz"
    assert unit.movement == 5
    assert unit.toughness == 5
    assert len(unit.ranged_weapons) == 1
    assert unit.ranged_weapons[0].name == "Shoota"

def test_parse_weapons_from_table():
    md = """| Weapon | Range | A | BS | S | AP | D | Abilities |
| Shoota | 18" | 3 | 5+ | 4 | 0 | 1 | Assault |
| Choppa | Melee | 3 | 4+ | 4 | 0 | 1 | - |"""
    ranged, melee = _parse_weapons_from_markdown(md)
    assert len(ranged) == 1
    assert len(melee) == 1
    assert ranged[0].name == "Shoota"
    assert melee[0].name == "Choppa"

def test_registry_cache(tmp_path):
    wiki_path = tmp_path / "wiki"
    (wiki_path / "units/orks").mkdir(parents=True)
    (wiki_path / "units/orks/Boyz.md").write_text(SAMPLE_BOY)
    reg = WikiRegistry(str(wiki_path))
    units = reg.load(use_cache=False)
    assert "Boyz" in units
