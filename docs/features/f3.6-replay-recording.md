# F3.6: Replay Recording — JSON Event Log per Round/Phase

**Файл:** `backend/engine/replay.py` (создать)
**Зависимости:** F2.5 (Game Loop), F3.5 (Auto-play), F2.1 (Game State)
**Часы:** 3h

**Цель (SMART):** Реализовать систему записи реплея: каждый ход/фаза/событие логируется в структурированный JSON с timestamp'ами, сериализуется и сохраняется в SQLite. Критерий: ReplayEncoder сериализует любой RoundLog в JSON без потерь (roundtrip test), SQLite хранит реплеи по game_id с возможностью загрузки по ID. 3 часа.

---

## Data Model

```python
from dataclasses import dataclass, field, asdict
from datetime import datetime
from typing import Optional
import json
import sqlite3


@dataclass
class ReplayEvent:
    """Одно событие в реплее."""
    timestamp: float  # ms от начала игры
    round: int
    phase: str
    turn: int
    event_type: str         # "shoot", "charge", "move", "kill", "damage", "cp_spend", etc.
    actor_id: str           # unit_id инициатора
    actor_name: str         # удобочитаемое имя
    target_id: Optional[str] = None
    target_name: Optional[str] = None
    weapon_index: Optional[int] = None
    weapon_name: Optional[str] = None
    dice_rolled: Optional[list[int]] = None
    result_value: Optional[float] = None  # урон, CP, VP
    detail: Optional[str] = None          # дополнительный текст
    position_before: Optional[dict] = None
    position_after: Optional[dict] = None


@dataclass
class ReplayRound:
    """Один раунд реплея."""
    round: int
    start_state: dict       # снимок GameState в начале раунда
    end_state: dict         # снимок в конце
    events: list[ReplayEvent] = field(default_factory=list)
    phase_summary: dict[str, dict] = field(default_factory=dict)


@dataclass
class Replay:
    """Полный реплей игры."""
    game_id: str            # UUID
    created_at: str         # ISO timestamp
    rosters: dict           # roster_a, roster_b
    mission: str
    deployment: str
    seed: int
    rounds: list[ReplayRound]
    summary: dict           # итоги (кто победил, счёт)
    version: str = "1.0"
```

## Replay Recorder

```python
class ReplayRecorder:
    """Записывает события игры в структурированный реплей."""

    def __init__(self, game_id: str, rosters: dict,
                 mission: str, deployment: str, seed: int):
        self.replay = Replay(
            game_id=game_id,
            created_at=datetime.utcnow().isoformat(),
            rosters=rosters,
            mission=mission,
            deployment=deployment,
            seed=seed,
            rounds=[],
            summary={},
        )
        self._start_time: Optional[float] = None
        self._current_round: Optional[ReplayRound] = None

    def start_game(self):
        """Запустить таймер реплея."""
        self._start_time = __import__('time').time()

    def start_round(self, round_num: int, state: GameState):
        """Начать новый раунд."""
        self._current_round = ReplayRound(
            round=round_num,
            start_state=_snapshot_state(state),
            end_state={},
        )

    def end_round(self, state: GameState):
        """Завершить раунд и сохранить снимок."""
        if self._current_round:
            self._current_round.end_state = _snapshot_state(state)
            self.replay.rounds.append(self._current_round)
            self._current_round = None

    def record(self, event: ReplayEvent):
        """Записать одно событие."""
        if self._start_time:
            elapsed = (__import__('time').time() - self._start_time) * 1000
            event.timestamp = elapsed
        if self._current_round is not None:
            self._current_round.events.append(event)

    def record_shoot(self, round_num: int, phase: str, turn: int,
                     actor: UnitState, target: UnitState,
                     weapon_name: str, damage: float,
                     dice: list[int] = None):
        """Записать событие стрельбы."""
        self.record(ReplayEvent(
            round=round_num, phase=phase, turn=turn,
            event_type="shoot",
            actor_id=actor.unit_id, actor_name=actor.unit_name,
            target_id=target.unit_id, target_name=target.unit_name,
            weapon_name=weapon_name,
            dice_rolled=dice,
            result_value=damage,
            position_before=_pos_dict(actor.position),
        ))

    def record_charge(self, round_num: int, phase: str, turn: int,
                      actor: UnitState, target: UnitState,
                      charge_roll: int, success: bool):
        """Записать событие заряда."""
        self.record(ReplayEvent(
            round=round_num, phase=phase, turn=turn,
            event_type="charge",
            actor_id=actor.unit_id, actor_name=actor.unit_name,
            target_id=target.unit_id, target_name=target.unit_name,
            dice_rolled=[charge_roll],
            result_value=1.0 if success else 0.0,
            detail="success" if success else "failed",
        ))

    def record_kill(self, round_num: int, phase: str, turn: int,
                    actor: UnitState, target: UnitState):
        self.record(ReplayEvent(
            round=round_num, phase=phase, turn=turn,
            event_type="kill",
            actor_id=actor.unit_id, actor_name=actor.unit_name,
            target_id=target.unit_id, target_name=target.unit_name,
            result_value=1.0,
        ))

    def record_move(self, round_num: int, phase: str, turn: int,
                    actor: UnitState,
                    from_pos: tuple, to_pos: tuple):
        self.record(ReplayEvent(
            round=round_num, phase=phase, turn=turn,
            event_type="move",
            actor_id=actor.unit_id, actor_name=actor.unit_name,
            position_before={"x": from_pos[0], "y": from_pos[1]},
            position_after={"x": to_pos[0], "y": to_pos[1]},
        ))

    def set_summary(self, summary: dict):
        self.replay.summary = summary


def _snapshot_state(state: GameState) -> dict:
    """Создать снимок GameState для реплея."""
    return {
        "round": state.round,
        "phase": state.phase,
        "turn": state.turn,
        "victory_points": dict(state.victory_points),
        "units": {
            "roster_a": [_unit_snapshot(u) for u in state.roster_a.units],
            "roster_b": [_unit_snapshot(u) for u in state.roster_b.units],
        },
    }


def _unit_snapshot(unit: UnitState) -> dict:
    return {
        "id": unit.unit_id,
        "name": unit.unit_name,
        "squad_size": unit.squad_size,
        "wounds_remaining": unit.wounds_remaining,
        "position": {"x": unit.position.x, "y": unit.position.y},
        "is_engaged": unit.is_engaged,
        "is_battle_shocked": unit.is_battle_shocked,
    }


def _pos_dict(pos) -> dict:
    return {"x": pos.x, "y": pos.y}
```

## Serialization

```python
class ReplayEncoder(json.JSONEncoder):
    """JSON encoder для Replay dataclasses."""
    def default(self, obj):
        if hasattr(obj, '__dataclass_fields__'):
            return asdict(obj)
        return super().default(obj)


def replay_to_json(replay: Replay) -> str:
    """Сериализовать Replay в JSON."""
    return json.dumps(asdict(replay), cls=ReplayEncoder, indent=2)


def replay_from_json(data: str) -> Replay:
    """Десериализовать Replay из JSON."""
    raw = json.loads(data)
    return _replay_from_dict(raw)


def _replay_from_dict(raw: dict) -> Replay:
    """Рекурсивно восстановить Replay из словаря."""
    rounds = []
    for r in raw.get("rounds", []):
        events = [ReplayEvent(**e) for e in r.get("events", [])]
        rounds.append(ReplayRound(
            round=r["round"],
            start_state=r.get("start_state", {}),
            end_state=r.get("end_state", {}),
            events=events,
            phase_summary=r.get("phase_summary", {}),
        ))

    return Replay(
        game_id=raw["game_id"],
        created_at=raw["created_at"],
        rosters=raw.get("rosters", {}),
        mission=raw.get("mission", ""),
        deployment=raw.get("deployment", ""),
        seed=raw.get("seed", 0),
        rounds=rounds,
        summary=raw.get("summary", {}),
        version=raw.get("version", "1.0"),
    )
```

## SQLite Persistence

```python
TABLE_REPLAYS = """
CREATE TABLE IF NOT EXISTS replays (
    game_id TEXT PRIMARY KEY,
    created_at TEXT NOT NULL,
    roster_a TEXT NOT NULL,
    roster_b TEXT NOT NULL,
    mission TEXT NOT NULL,
    deployment TEXT NOT NULL,
    seed INTEGER NOT NULL,
    replay_json TEXT NOT NULL,
    summary TEXT,
    user_id INTEGER,
    FOREIGN KEY (user_id) REFERENCES users(id)
)
"""


def save_replay(db: sqlite3.Connection, replay: Replay,
                user_id: Optional[int] = None):
    """Сохранить реплей в SQLite."""
    db.execute(
        """INSERT OR REPLACE INTO replays
           (game_id, created_at, roster_a, roster_b,
            mission, deployment, seed, replay_json, summary, user_id)
           VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)""",
        (
            replay.game_id,
            replay.created_at,
            json.dumps(replay.rosters.get("roster_a", {})),
            json.dumps(replay.rosters.get("roster_b", {})),
            replay.mission,
            replay.deployment,
            replay.seed,
            replay_to_json(replay),
            json.dumps(replay.summary),
            user_id,
        ),
    )
    db.commit()


def load_replay(db: sqlite3.Connection, game_id: str) -> Optional[Replay]:
    """Загрузить реплей из SQLite по game_id."""
    row = db.execute(
        "SELECT replay_json FROM replays WHERE game_id = ?",
        (game_id,),
    ).fetchone()
    if row:
        return replay_from_json(row[0])
    return None


def list_replays(db: sqlite3.Connection,
                 user_id: Optional[int] = None,
                 limit: int = 20) -> list[dict]:
    """Список реплеев (метаданные без полного JSON)."""
    if user_id:
        rows = db.execute(
            """SELECT game_id, created_at, mission, deployment,
                      seed, summary
               FROM replays WHERE user_id = ?
               ORDER BY created_at DESC LIMIT ?""",
            (user_id, limit),
        ).fetchall()
    else:
        rows = db.execute(
            """SELECT game_id, created_at, mission, deployment,
                      seed, summary
               FROM replays ORDER BY created_at DESC LIMIT ?""",
            (limit,),
        ).fetchall()

    return [
        {
            "game_id": r[0], "created_at": r[1],
            "mission": r[2], "deployment": r[3],
            "seed": r[4], "summary": json.loads(r[5]) if r[5] else {},
        }
        for r in rows
    ]
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Replay с 0 событий | Сохраняется, rounds=[], не падает |
| Огромный реплей (10K+ событий) | Сериализуется, SQLite BLOB не более 1MB |
| Спецсимволы в unit_name | Корректный JSON escape |
| Nested dataclasses (Replay → ReplayRound → ReplayEvent) | Рекурсивная сериализация через asdict |
| Загрузка несуществующего game_id | Вернуть None |
| Replay версии 1.0 vs future 2.0 | Поле version для обратной совместимости |
| Дублирование game_id | INSERT OR REPLACE (update) |

def test_replay_roundtrip():
    """Сериализация/десериализация Replay без потерь."""
    replay = make_test_replay()
    json_str = replay_to_json(replay)
    restored = replay_from_json(json_str)
    assert restored.game_id == replay.game_id
    assert len(restored.rounds) == len(replay.rounds)
    assert restored.rounds[0].events[0].event_type == "shoot"


def test_recorder_records_events():
    recorder = ReplayRecorder(
        game_id="test-1", rosters={},
        mission="only_war", deployment="standard", seed=42,
    )
    recorder.start_game()
    recorder.start_round(1, make_state())
    recorder.record_shoot(1, "shooting", 1, actor, target, "Shoota", 3.0)
    recorder.end_round(make_state())
    assert len(recorder.replay.rounds) == 1
    assert len(recorder.replay.rounds[0].events) == 1


def test_save_and_load_replay():
    """SQLite save → load возвращает тот же объект."""
    db = sqlite3.connect(":memory:")
    db.execute(TABLE_REPLAYS)
    replay = make_test_replay()
    save_replay(db, replay, user_id=1)
    loaded = load_replay(db, replay.game_id)
    assert loaded is not None
    assert loaded.game_id == replay.game_id
    assert len(loaded.rounds) == len(replay.rounds)


def test_load_nonexistent_replay():
    db = sqlite3.connect(":memory:")
    db.execute(TABLE_REPLAYS)
    result = load_replay(db, "nonexistent")
    assert result is None


def test_list_replays():
    db = sqlite3.connect(":memory:")
    db.execute(TABLE_REPLAYS)
    save_replay(db, make_test_replay(game_id="g1"), user_id=1)
    save_replay(db, make_test_replay(game_id="g2"), user_id=1)
    replays = list_replays(db, user_id=1)
    assert len(replays) == 2


def test_empty_replay():
    """Реплей без событий не падает при roundtrip."""
    replay = Replay(
        game_id="empty", created_at="2026-01-01",
        rosters={}, mission="", deployment="", seed=0,
        rounds=[], summary={},
    )
    json_str = replay_to_json(replay)
    restored = replay_from_json(json_str)
    assert len(restored.rounds) == 0


def test_recorder_timestamp():
    """События имеют возрастающие timestamp'ы."""
    recorder = ReplayRecorder(...)
    recorder.start_game()
    recorder.start_round(1, make_state())
    recorder.record(ReplayEvent(...))
    recorder.record(ReplayEvent(...))
    ts = [e.timestamp for e in recorder.replay.rounds[0].events]
    assert all(ts[i] <= ts[i+1] for i in range(len(ts)-1))
