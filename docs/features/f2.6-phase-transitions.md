# F2.6: Phase Transitions — Priority, Alternating Activations

**Файл:** `backend/engine/scenario.py` (расширение)
**Зависимости:** F2.5 (Game Loop)
**Часы:** 4h

**Цель (SMART):** Реализовать механизм чередования активаций в фазе Fight и определение Command Priority. Критерий: в Fight фазе игроки активируют по одному юниту, начиная с того, кто не ходил первым; Command Priority проверяется в начале каждого раунда. 4 часа.

## Fight Phase — Alternating Activations

```python
def fight_phase(state: GameState, map_: GameMap, log: RoundLog) -> GameState:
    """Fight Phase: alternating melee activations."""

    # Кто начинает: игрок, НЕ имеющий приоритет (т.е. ходил вторым в раунде)
    first_fighter = 2 if state.roster_a.command_priority else 1
    players = [first_fighter, 3 - first_fighter]

    # Собрать все engaged юниты
    engaged_units = []
    for player in players:
        roster = state.roster_a if player == 1 else state.roster_b
        for unit in roster.units:
            if unit.is_engaged and not unit.is_fighting:
                engaged_units.append((player, unit))

    # Alternating activations
    while any(u.squad_size > 0 and not u.is_fighting
              for _, u in engaged_units):
        for player in players:
            roster = state.roster_a if player == 1 else state.roster_b
            active = _select_fighter(roster, player)
            if active is None:
                continue
            # Resolve melee
            opponent = _find_melee_target(active, roster, state)
            if opponent:
                _resolve_melee(active, opponent, state, log)
            active.is_fighting = True

    # Reset flags
    for p in [1, 2]:
        roster = state.roster_a if p == 1 else state.roster_b
        for u in roster.units:
            u.is_fighting = False

    return state


def _select_fighter(roster: RosterState, player: int) -> Optional[UnitState]:
    """Выбрать юнит для активации в Fight."""
    choices = [u for u in roster.units
               if u.is_engaged and not u.is_fighting and u.squad_size > 0]
    return choices[0] if choices else None
```

## Command Priority

```python
def determine_command_priority(state: GameState) -> GameState:
    """В начале раунда определить, кто ходит первым.

    Leviathan rules:
    - Round 1: roll-off (D6, higher wins)
    - Round 2+: alternating (whoever went 2nd last round goes 1st now)
    """
    if state.battle_round == 1:
        # Roll-off
        roll_a = np.random.default_rng(state.seed).integers(1, 7)
        roll_b = np.random.default_rng(state.seed + 1).integers(1, 7)
        if roll_a >= roll_b:
            state.roster_a.command_priority = True
            state.roster_b.command_priority = False
        else:
            state.roster_a.command_priority = False
            state.roster_b.command_priority = True
        state.turn = 1 if state.roster_a.command_priority else 2
    else:
        # Swap
        old_p1 = state.roster_a.command_priority
        state.roster_a.command_priority = not old_p1
        state.roster_b.command_priority = old_p1
        state.turn = 1 if state.roster_a.command_priority else 2

    return state
```

## Activation Reset

```python
def _reset_activation_flags(state: GameState) -> None:
    """Сбросить флаги активации на начало раунда."""
    for roster in [state.roster_a, state.roster_b]:
        for unit in roster.units:
            unit.is_shooting = False
            unit.is_charging = False
            unit.is_fighting = False
            unit.is_advancing = False
            unit.cp_generated_this_turn = False
```
