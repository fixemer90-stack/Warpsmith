# F2.12: Leader Compatibility Checker

**Файл:** `backend/state/leader_checker.py` (создать)
**Зависимости:** F2.9 (Roster Validation), F1.1 (Unit)
**Часы:** 3h

**Цель (SMART):** Реализовать проверку совместимости лидеров: Character может присоединиться к отряду, если unit.name есть в leader_for лидера, и только один лидер на отряд с учётом Captain/Lieutenant rule. Критерий: Warboss + Boyz = OK, два Warboss на Boyz = ошибка, лидер без leader_for не может присоединиться. 3 часа.

## Core Logic

```python
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class LeaderCompatibilityResult:
    is_compatible: bool
    issues: list[str] = field(default_factory=list)

    @classmethod
    def ok(cls) -> "LeaderCompatibilityResult":
        return cls(is_compatible=True)

    @classmethod
    def fail(cls, *issues: str) -> "LeaderCompatibilityResult":
        return cls(is_compatible=False, issues=list(issues))
```

## Compatibility Rules

```python
def check_leader_compatibility(
    leader: Unit,
    target_unit: Unit,
    existing_leaders: list[Unit] = None,
) -> LeaderCompatibilityResult:
    """Проверить, может ли лидер присоединиться к отряду."""

    # Rule 1: Leader must have is_leader=True
    if not leader.is_leader:
        return LeaderCompatibilityResult.fail(
            f"{leader.name} is not a leader"
        )

    # Rule 2: Target must be in leader.leader_for list
    if not _is_compatible_unit(leader, target_unit):
        return LeaderCompatibilityResult.fail(
            f"{leader.name} cannot lead {target_unit.name}"
        )

    # Rule 3: Max 2 leaders per unit (Captain + Lieutenant rule)
    if existing_leaders and len(existing_leaders) >= 2:
        return LeaderCompatibilityResult.fail(
            f"Max 2 leaders per unit (already has {len(existing_leaders)})"
        )

    # Rule 4: Only 1 "Captain" type leader (not both can lead)
    if existing_leaders:
        leader_types = {_get_leader_type(l) for l in existing_leaders}
        new_type = _get_leader_type(leader)
        if new_type in leader_types:
            return LeaderCompatibilityResult.fail(
                f"{leader.name} cannot join: unit already has a {new_type}"
            )

    return LeaderCompatibilityResult.ok()


def _is_compatible_unit(leader: Unit, target: Unit) -> bool:
    """Проверка leader_for списка."""
    for allowed in leader.leader_for:
        if allowed == target.name:
            return True
        # Wildcard: "BATTLELINE" keyword
        if allowed.upper() == "BATTLELINE" and "Battleline" in target.keywords:
            return True
        if allowed.upper() == "INFANTRY" and "Infantry" in target.keywords:
            return True
    return False


def _get_leader_type(leader: Unit) -> str:
    """Определить тип лидера (Captain / Lieutenant / Other)."""
    keywords = [k.lower() for k in leader.keywords + leader.faction_keywords]
    if "captain" in leader.name.lower():
        return "captain"
    if "lieutenant" in leader.name.lower():
        return "lieutenant"
    return "other"
```

```python
def get_leader_hints(unit: Unit, all_units: dict[str, Unit]) -> list[str]:
    """Подсказка: какие лидеры подходят к юниту."""
    hints = []
    for name, candidate in all_units.items():
        if not candidate.is_leader:
            continue
        if _is_compatible_unit(candidate, unit):
            hints.append(f"{candidate.name} can lead this unit")
    return hints
```

## Тесты

```python
def test_leader_can_attach():
    warboss = Unit(name="Warboss", is_leader=True, leader_for=["Boyz"], ...)
    boyz = Unit(name="Boyz", keywords=["Infantry", "Battleline"], ...)
    result = check_leader_compatibility(warboss, boyz)
    assert result.is_compatible

def test_leader_wrong_unit():
    warboss = Unit(name="Warboss", is_leader=True, leader_for=["Boyz"], ...)
    marine = Unit(name="Tactical Marine", keywords=["Infantry"], ...)
    result = check_leader_compatibility(warboss, marine)
    assert not result.is_compatible
    assert "cannot lead" in result.issues[0]

def test_not_a_leader():
    boyz = Unit(name="Boyz", is_leader=False, ...)
    boyz2 = Unit(name="Boyz", ...)
    result = check_leader_compatibility(boyz, boyz2)
    assert not result.is_compatible
    assert "not a leader" in result.issues[0]

def test_max_two_leaders():
    warboss = Unit(name="Warboss", is_leader=True, leader_for=["Boyz"],
                   keywords=["Infantry", "Orks", "Character"], ...)
    weirdboy = Unit(name="Weirdboy", is_leader=True, leader_for=["Boyz"],
                    keywords=["Infantry", "Orks", "Psyker", "Character"], ...)
    nob = Unit(name="Nob", is_leader=True, leader_for=["Boyz"],
               keywords=["Infantry", "Orks", "Character"], ...)
    boyz = Unit(name="Boyz", keywords=["Infantry", "Battleline"], ...)
    result = check_leader_compatibility(nob, boyz, existing_leaders=[warboss, weirdboy])
    assert not result.is_compatible
    assert "Max 2 leaders" in result.issues[0]

def test_wildcard_battleline():
    """Лидер с leader_for=['BATTLELINE'] может присоединиться к любому Battleline."""
    leader = Unit(name="Generic Leader", is_leader=True, leader_for=["BATTLELINE"], ...)
    boyz = Unit(name="Boyz", keywords=["Battleline"], ...)

    assert result.is_compatible
```
