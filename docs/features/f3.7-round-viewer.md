# F3.7: Round Viewer — Step-by-Step Replay UI

**Файл:** `web/templates/round_viewer.html` (переписать) + `web/static/replay_viewer.js` (создать)
**Зависимости:** F3.6 (Replay Recording), F1.11 (Round Viewer Stub), F2.2 (2D Map)
**Часы:** 6h

**Цель (SMART):** Переработать Round Viewer в полноценный step-by-step replay просмотрщик: загрузка реплея по game_id, навигация по раундам/фазам/событиям, визуализация карты и юнитов на Canvas, подсветка активного действия. Критерий: пользователь может загрузить реплей, пройти все 5 раундов пошагово и увидеть каждое событие с визуализацией на карте. 6 часов.

---

## Data Flow

```
1. GET /replay/<game_id> → загрузить Replay из SQLite
2. Jinja2 рендерит round_viewer.html с метаданными
3. Alpine.js x-data="replayViewer()" управляет состоянием
4. Canvas отображает карту + юнитов
5. Step controls: ◀ Round ▼ Event ▶
6. Каждый шаг обновляет Canvas + info panel
```

## JavaScript (Alpine.js + Canvas)

```javascript
// web/static/replay_viewer.js

function replayViewer() {
    return {
        replay: null,
        currentRound: 1,
        eventIndex: 0,
        currentEvents: [],
        canvas: null,
        ctx: null,

        get currentPhase() {
            const round = this.replay?.rounds?.[this.currentRound - 1];
            return round?.events?.[this.eventIndex]?.phase || '—';
        },

        get currentTurn() {
            return this.replay?.rounds?.[this.currentRound - 1]
                       ?.events?.[this.eventIndex]?.turn || 1;
        },

        get currentEvent() {
            return this.currentEvents?.[this.eventIndex] || null;
        },

        get totalEvents() {
            return this.currentEvents?.length || 0;
        },

        get maxRounds() {
            return this.replay?.rounds?.length || 0;
        },

        get currentVP() {
            const round = this.replay?.rounds?.[this.currentRound - 1];
            const state = this.eventIndex === 0
                ? round?.start_state
                : round?.end_state;
            return state?.victory_points || {1: 0, 2: 0};
        },

        async loadReplay(gameId) {
            const resp = await fetch(`/api/replays/${gameId}`);
            this.replay = await resp.json();
            this.currentRound = 1;
            this.eventIndex = 0;
            this.updateEvents();
            this.$nextTick(() => this.initCanvas());
        },

        updateEvents() {
            const round = this.replay?.rounds?.[this.currentRound - 1];
            this.currentEvents = round?.events || [];
            if (this.eventIndex >= this.currentEvents.length) {
                this.eventIndex = Math.max(0, this.currentEvents.length - 1);
            }
            this.renderCanvas();
        },

        nextEvent() {
            if (this.eventIndex < this.totalEvents - 1) {
                this.eventIndex++;
                this.renderCanvas();
            }
        },

        prevEvent() {
            if (this.eventIndex > 0) {
                this.eventIndex--;
                this.renderCanvas();
            }
        },

        nextRound() {
            if (this.currentRound < this.maxRounds) {
                this.currentRound++;
                this.eventIndex = 0;
                this.updateEvents();
            }
        },

        prevRound() {
            if (this.currentRound > 1) {
                this.currentRound--;
                this.eventIndex = 0;
                this.updateEvents();
            }
        },

        initCanvas() {
            this.canvas = document.getElementById('replay-canvas');
            if (!this.canvas) return;
            this.ctx = this.canvas.getContext('2d');
            this.renderCanvas();
        },

        renderCanvas() {
            if (!this.ctx || !this.replay) return;
            const ctx = this.ctx;
            const w = this.canvas.width;
            const h = this.canvas.height;

            // Clear
            ctx.fillStyle = '#1a1a2e';
            ctx.fillRect(0, 0, w, h);

            // Grid
            ctx.strokeStyle = '#2a2a4e';
            ctx.lineWidth = 0.5;
            for (let x = 0; x <= w; x += 20) {
                ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, h); ctx.stroke();
            }
            for (let y = 0; y <= h; y += 20) {
                ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(w, y); ctx.stroke();
            }

            // Get state snapshot
            const round = this.replay.rounds[this.currentRound - 1];
            if (!round) return;
            const state = this.eventIndex === 0
                ? round.start_state
                : round.end_state;
            if (!state?.units) return;

            // Draw units
            this.drawUnits(state.units.roster_a, '#4ade80', ctx, w, h);
            this.drawUnits(state.units.roster_b, '#f87171', ctx, w, h);

            // Highlight active actor
            const event = this.currentEvent;
            if (event?.actor_id) {
                this.highlightUnit(event.actor_id, '#fbbf24', ctx, w, h);
            }
            if (event?.target_id) {
                this.highlightUnit(event.target_id, '#f87171', ctx, w, h);
            }
        },

        drawUnits(units, color, ctx, w, h) {
            for (const u of units) {
                const x = (u.position.x / 60) * w;
                const y = (u.position.y / 44) * h;
                const radius = Math.max(3, u.squad_size);

                ctx.beginPath();
                ctx.arc(x, y, radius, 0, Math.PI * 2);
                ctx.fillStyle = color;
                ctx.globalAlpha = u.squad_size > 0 ? 1.0 : 0.3;
                ctx.fill();
                ctx.globalAlpha = 1.0;

                // Label
                ctx.fillStyle = '#fff';
                ctx.font = '8px monospace';
                ctx.fillText(u.name.slice(0, 8), x + radius + 2, y + 3);
            }
        },

        highlightUnit(unitId, color, ctx, w, h) {
            const allUnits = [
                ...(this.replay.rounds[this.currentRound - 1]?.start_state?.units?.roster_a || []),
                ...(this.replay.rounds[this.currentRound - 1]?.start_state?.units?.roster_b || []),
            ];
            const unit = allUnits.find(u => u.id === unitId);
            if (!unit) return;

            const x = (unit.position.x / 60) * w;
            const y = (unit.position.y / 44) * h;
            ctx.beginPath();
            ctx.arc(x, y, 10, 0, Math.PI * 2);
            ctx.strokeStyle = color;
            ctx.lineWidth = 3;
            ctx.stroke();
        },

        eventTypeClass(event) {
            const map = {
                'shoot': 'bg-red-100 text-red-800',
                'charge': 'bg-orange-100 text-orange-800',
                'move': 'bg-blue-100 text-blue-800',
                'kill': 'bg-red-200 text-red-900 font-bold',
                'damage': 'bg-yellow-100 text-yellow-800',
                'cp_spend': 'bg-purple-100 text-purple-800',
            };
            return map[event?.event_type] || 'bg-gray-100 text-gray-800';
        },
    };
}
```

```python
# web/routes/api_replays.py — добавлено

@router.get("/api/replays")
async def list_replays(db: sqlite3.Connection = Depends(get_db),
                       user: Optional[User] = Depends(get_current_user)):
    """Список доступных реплеев."""
    replays = list_replays(db, user_id=user.id if user else None)
    return {"replays": replays}


@router.get("/api/replays/{game_id}")
async def get_replay(game_id: str,
                     db: sqlite3.Connection = Depends(get_db)):
    """Получить полный реплей по game_id."""
    replay = load_replay(db, game_id)
    if not replay:
        raise HTTPException(status_code=404, detail="Replay not found")
    return asdict(replay)


@router.get("/replay/{game_id}")
async def replay_page(game_id: str,
                      request: Request,
                      db: sqlite3.Connection = Depends(get_db)):
    """HTML страница просмотра реплея."""
    replay = load_replay(db, game_id)
    if not replay:
        raise HTTPException(status_code=404, detail="Replay not found")
    return templates.TemplateResponse(
        "round_viewer.html",
        {"request": request, "game_id": game_id,
         "meta": replay.summary},
    )
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Replay с 0 событий в раунде | Показать "No events", Canvas с начальной позицией |
| unit_id не найден в highlight | Пропустить подсветку без ошибки |
| Canvas resize / responsive | CSS width: 100%, сохранять aspect ratio |
| Очень много юнитов (40+) | Canvas кластеризация или уменьшение radius |
| Быстрая навигация (spam кнопок) | Alpine.js очередь, обработка через $nextTick |
| Загрузка несуществующего game_id | Показать 404 страницу |
| Replay version mismatch | Попытаться отобразить, показать предупреждение |
