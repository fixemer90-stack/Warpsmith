## Ork-specific Scoring Overrides

```python
def ork_score_charge(actor: UnitState, target: UnitState,
                     ctx: EvaluationContext) -> float:
    """Orks получают бонус к заряду.

    - '+Ere We Go!: переброс 2D6 заряда
    - Waaagh! active: +1 к броску заряда
    - Чем больше Orks заряжают одновременно, тем выше score
    - Приоритет целей с низким OC (сместить с точки)
    """
    base = score_charge(actor, target, ctx)

    # 'Ere We Go! — переброс повышает вероятность успеха
    base *= 1.2  # ~+20% за переброс

    # Waaagh! активен
    if _is_waaagh_active(ctx.state):
        base *= 1.3  # +1 к заряду

    # Бонус за массовость
    nearby_allies = _count_nearby_allies(actor, ctx.state, radius=6)
    base *= (1.0 + 0.1 * min(nearby_allies, 5))

    # Штраф против тяжёлых целей с -1Dmg
    if _has_fnp_or_damage_reduction(target):
        base *= 0.7

    return base


def ork_score_shoot(actor: UnitState, target: UnitState,
                    weapon_idx: int, ctx: EvaluationContext) -> float:
    """Orks предпочитают Shoo… э, стрельба у Orks'ов так себе.

    - BS 5+ → низкая вероятность попадания
    - Большие отряды → Dakka! больше кубов = выше шанс
    - Предпочитать цели с плохим save
    """
    base = score_shoot(actor, target, weapon_idx, ctx)

    # Штраф за плохую баллистику
    if actor.bs >= 5:
        base *= 0.6

    # Бонус за размер отряда (More Dakka!)
    if actor.squad_size >= 10:
        base *= 1.3

    return base
```

## Waaagh! Management

```python
class WaaaghState(Enum):
    INACTIVE = "inactive"
    ACTIVE = "active"        # раунды 2-3
    EXPIRED = "expired"


def manage_waaagh(state: GameState, round_num: int, turn: int) -> WaaaghState:
    """Определить, активен ли Waaagh! в текущем раунде.

    Orks активируют Waaagh! в начале раунда 2 или 3, если:
    - Есть минимум 3 отряда в пределах 12" от врага
    - Хотя бы одна точка спорная (контестится)
    - Не активирован ранее
    """
    current = state.ork_waaagh_state

    if current == WaaaghState.EXPIRED:
        return WaaaghState.EXPIRED

    if current == WaaaghState.INACTIVE and round_num >= 2:
        # Проверка условий для Waaagh!
        ork_units = state.roster_a.units if turn == 1 else state.roster_b.units
        nearby_count = sum(
            1 for u in ork_units
            if _is_near_enemy(u, state, distance=12)
        )
        if nearby_count >= 3:
            return WaaaghState.ACTIVE

    if current == WaaaghState.ACTIVE and round_num >= 4:
        return WaaaghState.EXPIRED

    return current
```

## Stratagem Usage

```python
def ork_use_stratagems(state: GameState, actor: UnitState,
                       action: Action) -> Optional[str]:
    """Orks используют стратагемы:

    - 'Ere We Go! (2 CP): переброс заряда, если бросок < 7
    - 'Ard as Nails (1 CP): +1 к save против AP-1/AP-2 оружия
    - Careen! (1 CP): взрыв транспортного средства при уничтожении
    - Insane Bravery (1 CP): авто-пас Battle-shock
    """
    if action.type == ActionType.CHARGE:
        # 'Ere We Go! стоит 2 CP
        if state.active_roster().cp >= 2:
            return "ere_we_go"

    if action.type == ActionType.MOVE and action.score > 0:
        if _is_going_to_be_shot(actor, state) and state.active_roster().cp >= 1:
            return "ard_as_nails"

    return None
```

## OrkAI Class

```python
class OrkAI:
    """Ork AI — агрессивная стратегия ближнего боя."""

    def __init__(self):
        self.weights = ORK_WEIGHTS
        self.waaagh: WaaaghState = WaaaghState.INACTIVE

    def decide(self, unit: UnitState, ctx: EvaluationContext) -> Action:
        """Принять решение за Ork-юнит."""
        # Проверить Waaagh!
        self.waaagh = manage_waaagh(ctx.state, ctx.state.round, ctx.turn)

        # Переопределить выбор с Ork-весами
        action = choose_action(unit, ctx, weights=self.weights)

        # Пост-обработка: если можно заряжать — заряжай
        charge_action = self._try_force_charge(unit, ctx, action)
        if charge_action:
            return charge_action

        # Применить стратагему
        strat = ork_use_stratagems(ctx.state, unit, action)
        if strat:
            ctx.state.active_roster().cp -= {"ere_we_go": 2,
                                             "ard_as_nails": 1}.get(strat, 1)
            action.rationale += f" +{strat}"

        return action

    def _try_force_charge(self, unit: UnitState, ctx: EvaluationContext,
                          original: Action) -> Optional[Action]:
        """Если в charge range — заряжаем (Ork приоритет)."""
        if ctx.phase != "charge":
            return None
        for target in ctx.opponent.units:
            if target.squad_size <= 0:
                continue
            dist = _distance(unit.position, target.position)
            if dist <= 12:
                return Action(
                    type=ActionType.CHARGE,
                    target_id=target.unit_id,
                    score=ork_score_charge(unit, target, ctx),
                    rationale="Waaagh! Charge!",
                )
        return None
```

## Edge Cases

| Edge Case | Ожидаемое поведение |
|-----------|---------------------|
| Waaagh! уже использован | Не активировать повторно, вернуть EXPIRED |
| CP < 2, нужно 'Ere We Go | Пропустить стратагему, заряжать без переброса |
| Все Ork отряды уничтожены | AI не делает ходов, game loop продолжается |
| Orks в транспорте | Не стреляют/не заряжаются, пока не высадятся |
| Waaagh! раунд 1 | Не активировать (только с раунда 2) |
| Нет стратагем в пуле | Не тратить CP |
| Цель с -1Dmg | Снизить score заряда на 30% |
| Ork отряд ниже половины | Не штрафовать — Orks не боятся |
| Противник с Overwatch | Заряжать всё равно (Ork mindset) |
| Большой отряд (20+ моделей) | Бонус к заряду + стратагемы |
