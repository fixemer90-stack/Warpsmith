# F4.10: Leaflet Map — Game Format + Mission Visualization

**Файлы:**
- `web/templates/partials/map_view.html` (замена `canvas_map.html`)
- `web/static/map_view.js` (замена `canvas_map.js`)
- `web/routes/api.py` (расширение `/api/map/tiles` — принять `width`/`height`)
- `web/templates/scenario_setup.html` (патч)
- `web/static/scenario_setup.js` (патч)

**Зависимости:** F2.2 (Map), F2.4 (Missions), F4.10 заменяет F4.5 (Canvas Map)
**Часы:** 6h

**Цель (SMART):** Заменить Canvas-карту на Leaflet.js с интерактивной сеткой, авто-масштабированием под формат игры, отображением миссий и фильтрацией ростеров по PTS. Критерий: без выбранного формата карта скрыта; после выбора — Leaflet рендерит сетку (44×30 … 44×90), objectives отображаются маркерами, deploy zones — цветными прямоугольниками, ростеры фильтруются по pts_limit ≤ формат. 6 часов.

---

## UI Flow

```
1. Пользователь открывает Scenario Setup
   → Карта СКРЫТА (плейсхолдер «Select game format to see the map»)
   → Список ростеров: все доступны (без фильтра)

2. Пользователь выбирает Game Format (500/1000/2000/3000)
   → Карта ПОЯВЛЯЕТСЯ: Leaflet с сеткой правильного размера
   → Mission objectives отображаются на карте
   → Deploy zones подсвечены
   → Ростеры фильтруются: только с pts_limit ≤ выбранный формат

3. Пользователь выбирает миссию
   → Objectives на карте обновляются

4. Пользователь выбирает ростеры
   → Юниты отображаются на карте, расстановка случайным образом в доступных местах на карте
```

```javascript
// map_view.js — Alpine.js component
function mapView() {
    return {
        map: null,
        tileLayer: null,
        objectivesLayer: null,
        deployZoneLayer: null,
        gridSize: { width: 0, height: 0 },

        initMap(width, height) {
            // Destroy previous map if exists
            if (this.map) { this.map.remove(); this.map = null; }

            const cellSize = 24; // px per cell

            this.gridSize = { width, height };
            this.map = L.map('battlefield-map', {
                crs: L.CRS.Simple,          // Flat grid, no geo projection
                minZoom: -2,
                maxZoom: 2,
                zoomControl: true,
                attributionControl: false,
            });

            // Set bounds to fit the entire grid
            const bounds = [[0, 0], [height * cellSize, width * cellSize]];
            this.map.setMaxBounds(bounds);
            this.map.fitBounds(bounds);

            // Draw grid cells with terrain
            this._drawTerrain(width, height, cellSize);
        },

        _drawTerrain(width, height, cellSize) {
            // Fetch tiles from API
            fetch(`/api/map/tiles?width=${width}&height=${height}`)
                .then(r => r.json())
                .then(data => this._renderTiles(data, cellSize));
        },

        _renderTiles(data, cellSize) {
            // Clear old layers
            if (this.tileLayer) this.map.removeLayer(this.tileLayer);

            const tiles = data.tiles;  // height × width matrix
            const features = [];

            for (let y = 0; y < tiles.length; y++) {
                for (let x = 0; x < tiles[y].length; x++) {
                    const color = TILE_COLORS[tiles[y][x]] || '#333';
                    features.push({
                        type: 'Feature',
                        geometry: {
                            type: 'Polygon',
                            coordinates: [[
                                [x * cellSize, y * cellSize],
                                [(x + 1) * cellSize, y * cellSize],
                                [(x + 1) * cellSize, (y + 1) * cellSize],
                                [x * cellSize, (y + 1) * cellSize],
                            ]],
                        },
                        properties: { color, tileType: tiles[y][x] },
                    });
                }
            }

            this.tileLayer = L.geoJSON(
                { type: 'FeatureCollection', features },
                {
                    style: (f) => ({
                        fillColor: f.properties.color,
                        color: '#1a1a2e',
                        weight: 0.5,
                        fillOpacity: 0.8,
                    }),
                }
            ).addTo(this.map);
        },

        showObjectives(objectives) {
            if (this.objectivesLayer) this.map.removeLayer(this.objectivesLayer);
            const markers = objectives.map(obj =>
                L.circleMarker([obj.y * 24 + 12, obj.x * 24 + 12], {
                    radius: 10,
                    color: '#fbbf24',
                    fillColor: '#fbbf24',
                    fillOpacity: 0.6,
                }).bindTooltip(obj.label || 'Objective')
            );
            this.objectivesLayer = L.layerGroup(markers).addTo(this.map);
        },

        showDeployZones(zones) {
            if (this.deployZoneLayer) this.map.removeLayer(this.deployZoneLayer);
            const rects = [];
            for (const [playerId, zoneCells] of Object.entries(zones)) {
                const color = playerId === 'p1' || playerId === '1' ? '#3b82f6' : '#ef4444';
                for (const [x, y] of zoneCells) {
                    rects.push(
                        L.rectangle(
                            [[y * 24, x * 24], [(y + 1) * 24, (x + 1) * 24]],
                            { color, fillColor: color, fillOpacity: 0.15, weight: 0.5 }
                        )
                    );
                }
            }
            this.deployZoneLayer = L.layerGroup(rects).addTo(this.map);
        },
    };
}
```

## Backend: /api/map/tiles — поддержка произвольных размеров

```python
@router.get("/map/tiles")
async def get_map_tiles(width: int = 16, height: int = 16):
    tiles = []
    for y in range(height):
        row = []
        for x in range(width):
            tile_type = 0  # OPEN — later: procedurally generate terrain
            row.append(tile_type)
        tiles.append(row)

    deploy_zones = {
        "p1": [(x, y) for x in range(width // 3) for y in range(height)],
        "p2": [(x, y) for x in range(2 * width // 3, width) for y in range(height)],
    }

    return {"width": width, "height": height, "tiles": tiles, "deploy_zones": deploy_zones}
```

```javascript
// scenario_setup.js
get compatibleRosters() {
    const pts = parseInt(this.gameFormat) || 2000;
    return this.rosters.filter(r => (r.pts_limit || 2000) <= pts);
}
```
